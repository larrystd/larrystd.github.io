> 基于g++ 4.8, 啥书都没源码来得是在

针对指针指向的对象, shared_ptr会有一个共享区记录该对象引用的计数, 当新增一个shared_ptr指向该对象, 引用计数会增加, 当减少一个指向该对象的shared_ptr, 引用计数会减少。当引用计数减少到0, 会析构该对象。共享区块持有维护对象的指针, shared_ptr只能读写对象无权析构对象, 析构对象权限在共享区块中。

除了shared_ptr使用引用计数use_count, weak_ptr也有一个引用计数weak_count, 只不过weak_count不控制对象的生存只有use_count控制。use_count和weak_count是放在一个区域维护的。

### 共享区块 shared_count weak_count
共享区块由`_Sp_counted_base`和维护, 二者构成继承关系, 后者实现了前者的部分函数(虚函数)。共享区块和对象应该是一对一的维护关系, 共享区块和shared_ptr对象是一对多的。
```cpp

template<_Lock_policy _Lp = __default_lock_policy>
    class _Sp_counted_base // 提供引用计数的原子性, 实际的共享区块.shared_count和weak_count
    : public _Mutex_base<_Lp>   //counted_base继承自mutex
    {
    public:  
      _Sp_counted_base() noexcept   // 初始化_M_weak_count=1
      : _M_use_count(1), _M_weak_count(1) { }
      
      virtual
      ~_Sp_counted_base() noexcept
      { }

      virtual void
      _M_dispose() noexcept = 0;    // 析构区块维护的对象, 被继承类实现
      
      // Called when _M_weak_count drops to zero.
      virtual void
      _M_destroy() noexcept
      { delete this; }  // 析构共享区块
      
      virtual void*
      _M_get_deleter(const std::type_info&) = 0;

      void
      _M_add_ref_copy() // 增加引用, 调用原子操作
      { __gnu_cxx::__atomic_add_dispatch(&_M_use_count, 1); }
  
      void
      _M_release() noexcept // 降低引用时调用
      {
        // Be race-detector-friendly.  For more info see bits/c++config.
        _GLIBCXX_SYNCHRONIZATION_HAPPENS_BEFORE(&_M_use_count);
	if (__gnu_cxx::__exchange_and_add_dispatch(&_M_use_count, -1) == 1) // return old value, 如果old value==1说明应该析构维护的对象, 即调用_M_dispose
	  {
            _GLIBCXX_SYNCHRONIZATION_HAPPENS_AFTER(&_M_use_count);  // 增加栅栏, happen_before防止优化乱序
	    _M_dispose(); // 这个作用是析构指向的对象
        // 当对象应该被析构时, _M_weak_count应该-1。如果use_count=0但weak_count不为0, 区块不会被析构, 只会析构维护的对象
	    if (_Mutex_base<_Lp>::_S_need_barriers)
	      {
	        _GLIBCXX_READ_MEM_BARRIER;
	        _GLIBCXX_WRITE_MEM_BARRIER;
	      }

            _GLIBCXX_SYNCHRONIZATION_HAPPENS_BEFORE(&_M_weak_count);
	    if (__gnu_cxx::__exchange_and_add_dispatch(&_M_weak_count,  //weak_count
						       -1) == 1)  // 调用release也就降weak_count如果_M_weak_count, old_value==1说明destroy整个区块
              {
                _GLIBCXX_SYNCHRONIZATION_HAPPENS_AFTER(&_M_weak_count);
	        _M_destroy();   // _M_weak_count=0, 析构整个区块
              }
	  }
      }
  
      void
      _M_weak_add_ref() noexcept    // weak_ptr使用, 增加一次_M_weak_count引用
      { __gnu_cxx::__atomic_add_dispatch(&_M_weak_count, 1); }

      void
      _M_weak_release() noexcept  // 减少一次weak_ptr引用
      {
        _GLIBCXX_SYNCHRONIZATION_HAPPENS_BEFORE(&_M_weak_count);
	if (__gnu_cxx::__exchange_and_add_dispatch(&_M_weak_count, -1) == 1)
	  {
            _GLIBCXX_SYNCHRONIZATION_HAPPENS_AFTER(&_M_weak_count);
	    if (_Mutex_base<_Lp>::_S_need_barriers)
	      {
	        // See _M_release(),
	        // destroy() must observe results of dispose()
	        _GLIBCXX_READ_MEM_BARRIER;
	        _GLIBCXX_WRITE_MEM_BARRIER;
	      }
	    _M_destroy();
	  }
      }
  
      long
      _M_get_use_count() const noexcept // 原子的获得use_count引用次数
      {

        return __atomic_load_n(&_M_use_count, __ATOMIC_RELAXED);
      }

    private:  
      _Sp_counted_base(_Sp_counted_base const&) = delete; // 不可拷贝构造
      _Sp_counted_base& operator=(_Sp_counted_base const&) = delete;

      _Atomic_word  _M_use_count;     // #shared 两个原子计数
      _Atomic_word  _M_weak_count;    // #weak + (#shared != 0)
    };

  template<typename _Ptr, _Lock_policy _Lp> // _Ptr是维护的对象指针
    class _Sp_counted_ptr final : public _Sp_counted_base<_Lp>
    {
    public:
      explicit
      _Sp_counted_ptr(_Ptr __p)
      : _M_ptr(__p) { }

      virtual void
      _M_dispose() noexcept // 析构_M_ptr, 即维护的对象
      { delete _M_ptr; }

      virtual void
      _M_destroy() noexcept
      { delete this; }

      virtual void*
      _M_get_deleter(const std::type_info&)
      { return 0; }

      _Sp_counted_ptr(const _Sp_counted_ptr&) = delete;
      _Sp_counted_ptr& operator=(const _Sp_counted_ptr&) = delete;

    protected:
      _Ptr             _M_ptr;  // copy constructor must not throw
    };
```

#### 指向共享区块的指针

shared_ptr对象会持有一个指向共享区块的指针, 当shared_ptr对象析构该指针操作共享区块引用-1。

该指针用shared_count对象维护, 如果想输入析构器, 将析构器作为变量参数即可, 析构器不是shared_count类型的一部分。

shared_count对象被析构时调用release让共享区块计数-1, shared_count执行拷贝构造, operator=时自动让共享区块+1
```cpp
template<_Lock_policy _Lp>
    class __shared_count  // shared_count内部维护的是_Sp_counted_base
    {
    public:

      template<typename _Ptr>
        explicit
	__shared_count(_Ptr __p) : _M_pi(0)
	{
	  __try
	    {
	      _M_pi = new _Sp_counted_ptr<_Ptr, _Lp>(__p);  /// 创建一个共享区块指向对象, 因此对于裸指针会创建一个shared_count区块, 如果两个裸指针指向同一对象, 这样就会有两个共享区块指向对象, 可能造成重复析构错误。因此必须保证一个区块维护一个对象一对一管
	    }
	}

  // shared_count维护指向共享变量的指针和析构对象deleter
      template<typename _Ptr, typename _Deleter>
	__shared_count(_Ptr __p, _Deleter __d) : _M_pi(0)   // shared_ptr计数维护的共享区块
	{
	  // The allocator's value_type doesn't matter, will rebind it anyway.
	  typedef std::allocator<int> _Alloc;
	  typedef _Sp_counted_deleter<_Ptr, _Deleter, _Alloc, _Lp> _Sp_cd_type; // _Ptr, deleter等类型封装的对象
	  typedef typename allocator_traits<_Alloc>::template
	    rebind_traits<_Sp_cd_type> _Alloc_traits;
	  typename _Alloc_traits::allocator_type __a;
	  _Sp_cd_type* __mem = 0;
	  __try
	    {
	      __mem = _Alloc_traits::allocate(__a, 1);
	      _Alloc_traits::construct(__a, __mem, __p, std::move(__d));  // 分配__p, 析构对象_Deleter的区块维护
	      _M_pi = __mem;    // _M_pi是新区块
	    }
	}
      ~__shared_count() noexcept  // shared_count被析构, 调用共享区块的_M_release()使计数减1
      {
	if (_M_pi != 0)
	  _M_pi->_M_release();  
      }

    // 拷贝构造, 自动增加计数
    __shared_count(const __shared_count& __r) noexcept
      : _M_pi(__r._M_pi)
      {
	if (_M_pi != 0)
	  _M_pi->_M_add_ref_copy();
      }

      __shared_count& // shared_count 执行operator=赋值时自动增加计数
      operator=(const __shared_count& __r) noexcept
      {
	_Sp_counted_base<_Lp>* __tmp = __r._M_pi;
	if (__tmp != _M_pi)
	  {
	    if (__tmp != 0)
	      __tmp->_M_add_ref_copy();
	    if (_M_pi != 0)
	      _M_pi->_M_release();
	    _M_pi = __tmp;
	  }
	return *this;
      }

      long
      _M_get_use_count() const noexcept // 返回共享区块计数
      { return _M_pi != 0 ? _M_pi->_M_get_use_count() : 0; }

      bool
      _M_unique() const noexcept    // 共享区块计数是否为1
      { return this->_M_get_use_count() == 1; }

      _Sp_counted_base<_Lp>*  _M_pi;  // shared_count是一个维护M_Pi, M_pi指向实际的共享计数区域, 即_Sp_counted_base
    };

```
同理, weak_ptr也是操作共享区块weak_count的指针, 只是这个weak_count不影响对象的生存, 只影响共享区块的生存。weak_count和shared_count一样都持有_Sp_counted_base的指针。利用这个指针__weak_count也能实现shared_count一样的功能, 但那是不被允许的，但当将weak_ptr转为shared_ptr会用到weak_count内部_Sp_counted_base的指针。

weak_count不可创建共享区块, 因此必定先有shared_ptr才有weak_ptr。

```cpp
template<_Lock_policy _Lp>
    class __weak_count
    {
    public:
      constexpr __weak_count() noexcept : _M_pi(0)
      { }
        // 用__shared_count构造weak_count, 增加weak_count
      __weak_count(const __shared_count<_Lp>& __r) noexcept
      : _M_pi(__r._M_pi)
      {
	if (_M_pi != 0)
	  _M_pi->_M_weak_add_ref();
      }
    // 用__weak_count构造weak_count, 增加weak_count计数
      __weak_count(const __weak_count<_Lp>& __r) noexcept
      : _M_pi(__r._M_pi)
      {
	if (_M_pi != 0)
	  _M_pi->_M_weak_add_ref();
      }
        // 析构减少weak计数
      ~__weak_count() noexcept
      {
	if (_M_pi != 0)
	  _M_pi->_M_weak_release();
      }

    /// operator=
      __weak_count<_Lp>&
      operator=(const __shared_count<_Lp>& __r) noexcept
      {
	_Sp_counted_base<_Lp>* __tmp = __r._M_pi;
	if (__tmp != 0)
	  __tmp->_M_weak_add_ref();
	if (_M_pi != 0)
	  _M_pi->_M_weak_release();
	_M_pi = __tmp;
	return *this;
      }

      long
      _M_get_use_count() const noexcept
      { return _M_pi != 0 ? _M_pi->_M_get_use_count() : 0; }

      bool
      _M_less(const __weak_count& __rhs) const noexcept
      { return std::less<_Sp_counted_base<_Lp>*>()(this->_M_pi, __rhs._M_pi); }

      bool
      _M_less(const __shared_count<_Lp>& __rhs) const noexcept
      { return std::less<_Sp_counted_base<_Lp>*>()(this->_M_pi, __rhs._M_pi); }

      // Friend function injected into enclosing namespace and found by ADL
      friend inline bool
      operator==(const __weak_count& __a, const __weak_count& __b) noexcept
      { return __a._M_pi == __b._M_pi; }

    private:
      friend class __shared_count<_Lp>;

      _Sp_counted_base<_Lp>*  _M_pi;
    };
```

### shared_ptr 和weak_ptr


#### shared_ptr
shared_ptr的实现主要在`bits/shared_ptr_base.h`和`shared_ptr.h`中。

显然shared_ptr应该有一个shared_count指向共享计数, 也需要维护一个指向对象的裸指针, 该裸指针只能读写对象, 不能执行析构对象。

```cpp
 template<typename _Tp, _Lock_policy _Lp>
    class __shared_ptr
    {
    public:
      typedef _Tp   element_type;

      template<typename _Tp1>
	explicit __shared_ptr(_Tp1* __p)  // 用裸指针构造shared_ptr对象， 为p构造shared_count, _M_ptr。这其实新增了一个共享区块, 千万注意防止多个共享区块维护同一个对象带来的问题
        : _M_ptr(__p), _M_refcount(__p)
	{
	  __glibcxx_function_requires(_ConvertibleConcept<_Tp1*, _Tp*>)
	  static_assert( sizeof(_Tp1) > 0, "incomplete type" );
	  __enable_shared_from_this_helper(_M_refcount, __p, __p);  // 调用__enable_shared_from_this_helper
	}

      template<typename _Tp1, typename _Deleter>
	__shared_ptr(_Tp1* __p, _Deleter __d) // 用裸指针和deleter来构造shared_ptr对象
	: _M_ptr(__p), _M_refcount(__p, __d)
	{
	  __glibcxx_function_requires(_ConvertibleConcept<_Tp1*, _Tp*>)
	  // TODO requires _Deleter CopyConstructible and __d(__p) well-formed
	  __enable_shared_from_this_helper(_M_refcount, __p, __p);
	}

      template<typename _Tp1, typename = typename
	       std::enable_if<std::is_convertible<_Tp1*, _Tp*>::value>::type> // std::enable_if满足条件时类型有效, 结果是拷贝构造_M_refcount, 共享区块计数+1
	__shared_ptr(const __shared_ptr<_Tp1, _Lp>& __r) noexcept
	: _M_ptr(__r._M_ptr), _M_refcount(__r._M_refcount)
	{ }
    // 移动语义相当于交换了两个_M_refcount, 共享区块引用次数不变
      __shared_ptr(__shared_ptr&& __r) noexcept
      : _M_ptr(__r._M_ptr), _M_refcount()
      {
	_M_refcount._M_swap(__r._M_refcount);
	__r._M_ptr = 0;
      }
    // 用__weak_ptr构造shared_ptr, 
      template<typename _Tp1>
	explicit __shared_ptr(const __weak_ptr<_Tp1, _Lp>& __r)
	: _M_refcount(__r._M_refcount) // may throw
	{
	  __glibcxx_function_requires(_ConvertibleConcept<_Tp1*, _Tp*>)

	  // It is now safe to copy __r._M_ptr, as
	  // _M_refcount(__r._M_refcount) did not throw.
	  _M_ptr = __r._M_ptr;  // 进行shared_count的操作
	}
    // 用unique_ptr构造shared_ptr, 需要新创建__shared_count。注意这里用的是移动语义, 也就是说unique_ptr不再维护对象, 转交给shared_ptr维护, 不存在同时维护一说。
      template<typename _Tp1, typename _Del>
	__shared_ptr(std::unique_ptr<_Tp1, _Del>&& __r)
	: _M_ptr(__r.get()), _M_refcount()
	{
	  __glibcxx_function_requires(_ConvertibleConcept<_Tp1*, _Tp*>)
	  auto __tmp = std::__addressof(*__r.get());
	  _M_refcount = __shared_count<_Lp>(std::move(__r));
	  __enable_shared_from_this_helper(_M_refcount, __tmp, __tmp);
	}
```

常用函数
```cpp
// reset, 重置
     void
      reset() noexcept  // 重置为__shared_ptr()空对象
      { __shared_ptr().swap(*this); }

      template<typename _Tp1>
	void
	reset(_Tp1* __p) // _Tp1 must be complete.  // 重置为__shared_ptr(__p)
	{
	  // Catch self-reset errors.
	  _GLIBCXX_DEBUG_ASSERT(__p == 0 || __p != _M_ptr);
	  __shared_ptr(__p).swap(*this);

      _Tp*
      operator->() const noexcept // 重载operator->, 实现指针->功能

      { return _M_ptr; }

      explicit operator bool() const // never throws
      { return _M_ptr == 0 ? false : true; }

      unique() const noexcept // 引用计数是否为1

      { return _M_refcount._M_get_use_count(); }

      swap(__shared_ptr<_Tp, _Lp>& __other) noexcept
      {
	std::swap(_M_ptr, __other._M_ptr);
	_M_refcount._M_swap(__other._M_refcount);
      }
```

#### weak_ptr

weak_ptr实现与shared_ptr类似, 只是内部维护weak_count, 不影响对象的生存周期。

```cpp
  template<typename _Tp, _Lock_policy _Lp>
    class __weak_ptr  // weak_ptr, 调用
    {
    public:
      typedef _Tp element_type;
    // 由weak_ptr构造
      template<typename _Tp1, typename = typename
	       std::enable_if<std::is_convertible<_Tp1*, _Tp*>::value>::type>
	__weak_ptr(const __weak_ptr<_Tp1, _Lp>& __r) noexcept
	: _M_refcount(__r._M_refcount)
        { _M_ptr = __r.lock().get(); }
    // 由shared_ptr构造
      template<typename _Tp1, typename = typename
	       std::enable_if<std::is_convertible<_Tp1*, _Tp*>::value>::type>
	__weak_ptr(const __shared_ptr<_Tp1, _Lp>& __r) noexcept
	: _M_ptr(__r._M_ptr), _M_refcount(__r._M_refcount)
	{ }

      template<typename _Tp1>
	__weak_ptr&
	operator=(const __weak_ptr<_Tp1, _Lp>& __r) noexcept
	{
	  _M_ptr = __r.lock().get();
	  _M_refcount = __r._M_refcount;
	  return *this;
	}
    // lock()自动有通过expired()判断是否生存, 如果存活生成shared_ptr, 不存活生成空的__shared_ptr

      __shared_ptr<_Tp, _Lp>
      lock() const noexcept
      {

	return expired() ? __shared_ptr<element_type, _Lp>()
			 : __shared_ptr<element_type, _Lp>(*this);

      long
      use_count() const noexcept
      { return _M_refcount._M_get_use_count(); }

    // 这个更方便, 判断是否存活。存活返回false
      bool
      expired() const noexcept  // 是否存活
      { return _M_refcount._M_get_use_count() == 0; }

      void
      reset() noexcept
      { __weak_ptr().swap(*this); }
```

### unique_ptr

unique_ptr在实现上与shared_ptr有很大不同, 主要体现在unqiue_ptr直接绑定维护的对象, 不需要生存共享区块来引用计数。unique_ptr适合单引用的对象, 比如数据库连接这种, 一个数据库连接对象一个引用。

与shared_ptr不同的, 析构器本身类似是unique_ptr的一部分。

unique_ptr不支持拷贝构造, 拷贝赋值, 但支持移动构造, 移动赋值。注意不要让多个unique_ptr维护同一个对象, 这样会导致重复析构段错误

```cpp
// 默认析构对象
  template<typename _Tp>
    struct default_delete // 默认析构对象
    {
      constexpr default_delete() noexcept = default;

      template<typename _Up, typename = typename
	       enable_if<is_convertible<_Up*, _Tp*>::value>::type>
        default_delete(const default_delete<_Up>&) noexcept { }

      void
      operator()(_Tp* __ptr) const
      {
	static_assert(sizeof(_Tp)>0,
		      "can't delete pointer to incomplete type");
	delete __ptr; // 调用delete ptr
      }
    };

// unique_ptr
  template <typename _Tp, typename _Dp = default_delete<_Tp> >
    class unique_ptr
    {
      // use SFINAE to determine whether _Del::pointer exists
// Pointer是一个指针对象
      class _Pointer
      {
	template<typename _Up>
	  static typename _Up::pointer __test(typename _Up::pointer*);

	template<typename _Up>
	  static _Tp* __test(...);

	typedef typename remove_reference<_Dp>::type _Del;

      public:
	typedef decltype(__test<_Del>(0)) type;
      };
// __tuple_type， _M_t是模板元组, 维护了指向对象的指针对象, 析构器_Dp
      typedef std::tuple<typename _Pointer::type, _Dp>  __tuple_type; // std::tuple可以看成泛型的pair
      __tuple_type                                      _M_t; // M_t时pointer::type, Dp的元组

// 构造器, 只传入裸指针则调用默认析构器
     explicit
      unique_ptr(pointer __p) noexcept
      : _M_t(__p, deleter_type())
      { static_assert(!is_pointer<deleter_type>::value,
		     "constructed with null function pointer deleter"); }

      unique_ptr(pointer __p,
	  typename conditional<is_reference<deleter_type>::value,
	    deleter_type, const deleter_type&>::type __d) noexcept
      : _M_t(__p, __d) { }

// Destructor.析构器, get_deleter()(__ptr);表示, 获取析构器get_deleter()得到函数, 然后调用函数(__ptr)析构之
      ~unique_ptr() noexcept  // 析构调用get_deleter
      {
	auto& __ptr = std::get<0>(_M_t);
	if (__ptr != nullptr)
	  get_deleter()(__ptr);
	__ptr = pointer();
      }
// 支持 operator=移动语义
    unique_ptr&
      operator=(unique_ptr&& __u) noexcept
      {
	reset(__u.release());
	get_deleter() = std::forward<deleter_type>(__u.get_deleter());
	return *this;
      }

// Move constructors.支持移动构造
      unique_ptr(unique_ptr&& __u) noexcept
      : _M_t(__u.release(), std::forward<deleter_type>(__u.get_deleter())) { }

	unique_ptr(unique_ptr<_Up, _Ep>&& __u) noexcept
	: _M_t(__u.release(), std::forward<_Ep>(__u.get_deleter()))
	{ }
// 不支持unique_ptr的拷贝构造, 拷贝赋值

// 可以赋值为nullptr,相当于reset()
      unique_ptr&
      operator=(nullptr_t) noexcept
      {
	reset();
	return *this;
      }

      // Observers.
      typename add_lvalue_reference<element_type>::type
      operator*() const
      {
	_GLIBCXX_DEBUG_ASSERT(get() != pointer());
	return *get();
      }

      pointer
      operator->() const noexcept
      {
	_GLIBCXX_DEBUG_ASSERT(get() != pointer());
	return get();
      }
// unique_ptr的pointer, deleter()都是turple _M_t维护的因此获取通过std::get<0>(_M_t)
      pointer
      get() const noexcept
      { return std::get<0>(_M_t); }

      deleter_type&
      get_deleter() noexcept
      { return std::get<1>(_M_t); }

// release 只是放弃持有对象的所有权, 返回裸指针, 不会析构对象,reset()才会析构对象。
            pointer
      release() noexcept
      {
	pointer __p = get();
	std::get<0>(_M_t) = pointer();
	return __p;
      }

      void
      reset(pointer __p = pointer()) noexcept
      {
	using std::swap;
	swap(std::get<0>(_M_t), __p);
	if (__p != pointer()) // 如果p和pointer不是一个对象, 则析构之
	  get_deleter()(__p);
      }
```

### 其他相关函数

#### enable_shared_from_this

前面说过, 使用shared_ptr要小心为同一个对象建立多个共享区块。而当我们函数返回一个shared_ptr时可能会如此。例如
```cpp
class Foo
: public enable_shared_from_this<Foo>
{
public:
    shared_ptr<Foo> returnSp()
    {
        return shared_ptr(this);// 这里对this指向的对象又创建了一个区块维护, 导致重复析构的风险
    }
};
```

解决办法时将`return shared_ptr(this)`改为`return shared_from_this(this)`
```cpp
class Foo
: public enable_shared_from_this<Foo>
{
public:
    shared_ptr<Foo> returnSp()
    {
        return shared_from_this();
    }
};
```

`shared_from_this`实际就是基于`weak_ptr`构造shared_ptr, 因此使用它的前提是`Foo`对象已经有`shared_ptr`维护了(有shared_ptr才能有共享区块才能有weak_ptr)。

```cpp
  template<typename _Tp>  // enable_shared_from_this, 防止生成多个计数块, 继承这个对象。可以调用shared_from_this函数
    class enable_shared_from_this
    {
    protected:
      constexpr enable_shared_from_this() noexcept { }

      enable_shared_from_this(const enable_shared_from_this&) noexcept { }
        // 赋值操作符, 返回的是引用, 防止a=b=c连续赋值情况, 返回值亦可作参数输入
      enable_shared_from_this&
      operator=(const enable_shared_from_this&) noexcept
      { return *this; }

      ~enable_shared_from_this() { }

    public:
      shared_ptr<_Tp>
      shared_from_this()  // 返回被维护的this指针, 相当于用weak_ptr构造shared_ptr, 因此不会另创建区块
      { return shared_ptr<_Tp>(this->_M_weak_this); }

    private:
      template<typename _Tp1>
	void
	_M_weak_assign(_Tp1* __p, const __shared_count<>& __n) const noexcept
	{ _M_weak_this._M_assign(__p, __n); }

      mutable weak_ptr<_Tp>  _M_weak_this;
    };
```

#### make_shared()

对一个泛型参数对象, 返回shared_ptr维护的对象。 make_shared的作用是, 不再需要任何裸指针就可以自动生成shared_ptr维护的对象, 从而减少可能出现裸指针的问题, 以及增加一步裸指针带来的效率开销。

```cpp
class A {
public:
    int a_;
    A(): a_(6){}
    A(int a): a_(a) {}
};

int main() {
    A* p = new A();
    auto sa = make_shared<A>();
    cout << sa->a_ <<endl;
    auto sb = make_shared<A>(5);
    cout << sb->a_ <<endl;
    return 0;
}
// 输出
6
5
```

实现
```cpp
  template<typename _Tp, typename... _Args>
    inline shared_ptr<_Tp>
    make_shared(_Args&&... __args)  // 使用make_shared, 返回std::allocate_shared<_Tp>
    {
      typedef typename std::remove_const<_Tp>::type _Tp_nc;
      return std::allocate_shared<_Tp>(std::allocator<_Tp_nc>(),
				       std::forward<_Args>(__args)...);
    }

  template<typename _Tp, typename _Alloc, typename... _Args>
    inline shared_ptr<_Tp> //基于参数构建Tp对象,不需要裸指针
    allocate_shared(const _Alloc& __a, _Args&&... __args)
    {
      // 返回一个shared_ptr
      return shared_ptr<_Tp>(_Sp_make_shared_tag(), __a,
			     std::forward<_Args>(__args)...);
    }
```


#### auto_ptr

auto_ptr的主要问题是进行copy, operator=转移指针资源的所有权的会同时将原指针置为NULL，这跟通常理解的copy行为是不一致的(不会修改原数据)，而这样的行为在有些场合下不是我们希望看到的。例如参考《Effective STL》第8条，sort的快排实现中有将元素复制到某个局部临时对象中，但对于auto_ptr，却将原元素置为null，这就导致最后的排序结果中可能有大量的null。而现在C++11的对move语义的支持，使得这样的资源转移通常只会在必要的场合发生，而auto_ptr是滥用的资源转移。

需要资源拷贝用shared_ptr, 资源转移或不改变资源所有权用unique_ptr。auto_ptr不支持与shared_ptr, unique_ptr的转换, 基本被弃用。
```cpp
    class auto_ptr
    {
    private:
      _Tp* _M_ptr;
      
    public:
      /// The pointed-to type.
      typedef _Tp element_type;
      
      explicit
      auto_ptr(element_type* __p = 0) throw() : _M_ptr(__p) { }

      auto_ptr(auto_ptr& __a) throw() : _M_ptr(__a.release()) { }

      template<typename _Tp1>
        auto_ptr(auto_ptr<_Tp1>& __a) throw() : _M_ptr(__a.release()) { }
// 执行operator=会释放原有资源, 这应该是&&而不是&
      auto_ptr&
      operator=(auto_ptr& __a) throw()
      {
	reset(__a.release());
	return *this;
      }


      ~auto_ptr() { delete _M_ptr; }
      
      element_type&
      operator*() const throw() 
      {
	__glibcxx_assert(_M_ptr != 0);
	return *_M_ptr; 
      }
      
      element_type*
      operator->() const throw() 
      {
	__glibcxx_assert(_M_ptr != 0);
	return _M_ptr; 
      }

      element_type*
      get() const throw() { return _M_ptr; }
      
      element_type*
      release() throw()
      {
	element_type* __tmp = _M_ptr;
	_M_ptr = 0;
	return __tmp;
      }
      
      void
      reset(element_type* __p = 0) throw()
      {
	if (__p != _M_ptr)
	  {
	    delete _M_ptr;
	    _M_ptr = __p;
	  }
```

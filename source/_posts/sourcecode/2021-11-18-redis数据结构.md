### 双向链表

故名思义，C实现的双向链表。

链表的优势是头插入, 删除比较简单, 但是查找比较耗时。

链表在Redis中可用来实现列表键, 发布与订阅, 慢查询, 监视器等。

常见函数的实现
```cpp
list *listCreate(void)  // 创建链表, 初始化成员, 返回list* 指针
{
    struct list *list;

    if ((list = zmalloc(sizeof(*list))) == NULL)
        return NULL;
    list->head = list->tail = NULL;
    list->len = 0;
    list->dup = NULL;
    list->free = NULL;
    list->match = NULL;
    return list;
}
list *listAddNodeHead(list *list, void *value)  // 头插法插入链表节点
{
    listNode *node;

    if ((node = zmalloc(sizeof(*node))) == NULL)
        return NULL;
    node->value = value;
    if (list->len == 0) {
        list->head = list->tail = node;
        node->prev = node->next = NULL;
    } else {
        node->prev = NULL;  // 头插法插入链表
        node->next = list->head;
        list->head->prev = node;
        list->head = node;
    }
    list->len++;
    return list;
}

listNode *listSearchKey(list *list, void *key)  // 查找一个key
{
    listIter iter;
    listNode *node;

    listRewind(list, &iter);
    while((node = listNext(&iter)) != NULL) {   // iter顺序查找
        if (list->match) {  // 如果匹配, 返回node
            if (list->match(node->value, key)) {
                return node;
            }
        } else {
            if (key == node->value) {
                return node;
            }
        }
    }
    return NULL;
}
```

### 动态字符串

redis的动态字符串SDS(simple dynamic string), 内部维护的就是`char*`, 外部对其长度等增加了限制。为了节省内存, 根据字符串的长度为其创建`sdshdr8`,`sdshdr16`, `sdshdr32`等不同长度的sds进行维护, 注意`sdshdr8`表示长度小于8个字节, 也就是小于256个字符。

**Redis使用SDS(Simple Dynamic String)作为字符串表示**

```cpp
typedef char *sds;  // 表示动态字符串内置成员弹性数组buf

// 8,16,32,64表示字符串的最大长度的位数。
struct __attribute__ ((__packed__)) sdshdr8 {       // 默认使用的sdshdr, 即uint8_t表示的长度可以是256, 基本够用了
    uint8_t len; /* used 已经使用数组的长度 */ 
    uint8_t alloc; /* excluding the header and null terminator sh->alloc - sh->len 表示可用空间*/
    unsigned char flags; /* 3 lsb of type, 5 unused bits m 表示type, 如SDS_TYPE_8*/
    char buf[]; // 保存字符串
};
struct __attribute__ ((__packed__)) sdshdr16 {
    uint16_t len; /* used */
    uint16_t alloc; /* excluding the header and null terminator */
    unsigned char flags; /* 3 lsb of type, 5 unused bits */
    char buf[];
};

#define SDS_TYPE_5  0 // 不同长度sds的代指
#define SDS_TYPE_8  1
#define SDS_TYPE_16 2
#define SDS_TYPE_32 3

#define SDS_HDR(T,s) ((struct sdshdr##T *)((s)-(sizeof(struct sdshdr##T))))  // s指向buf[]首地址, s-sizeof(..) 得到sdshdr32/sdshdr16 结构体首地址

static inline size_t sdslen(const sds s) {  // 对指向buf[]首地址的指针s, SDS_HDR可以返回sds对象的地址, 从而返回已经使用的长度
    unsigned char flags = s[-1];
    switch(flags&SDS_TYPE_MASK) {
        case SDS_TYPE_5:
            return SDS_TYPE_5_LEN(flags);
        case SDS_TYPE_8:
            return SDS_HDR(8,s)->len;
        ...
    }
    return 0;
}

static inline size_t sdsavail(const sds s) {    // sh->alloc-sh->len表示剩余可用空间
    unsigned char flags = s[-1];
    switch(flags&SDS_TYPE_MASK) {
        case SDS_TYPE_5: {
            return 0;
        }
        case SDS_TYPE_8: {
            SDS_HDR_VAR(8,s);
            return sh->alloc - sh->len; 
        }
        case SDS_TYPE_16: {
            SDS_HDR_VAR(16,s);
            return sh->alloc - sh->len;
        }
        ...
    }
    return 0;
}
```

```cpp
// 类似调用mystring = sdsnewlen("abc",3);
sds sdsnewlen(const void *init, size_t initlen) {   // len是buf[]数组的长度, 且正好等于*init数组长度, 返回对象指针sds
    void *sh;
    sds s;
    char type = sdsReqType(initlen);
    /* Empty strings are usually created in order to append. Use type 8
     * since type 5 is not good at this. */
    if (type == SDS_TYPE_5 && initlen == 0) type = SDS_TYPE_8;
    int hdrlen = sdsHdrSize(type);  // Hdr部分的size
    unsigned char *fp; /* flags pointer. */

    assert(hdrlen+initlen+1 > initlen); /* Catch size_t overflow */
    sh = s_malloc(hdrlen+initlen+1);    // 开辟空间, hdrlen+initlen+'\0'
    if (init==SDS_NOINIT)
        init = NULL;
    else if (!init) // 设置初始值
        memset(sh, 0, hdrlen+initlen+1);
    if (sh == NULL) return NULL;
    s = (char*)sh+hdrlen;
    fp = ((unsigned char*)s)-1;
    switch(type) {
        case SDS_TYPE_8: {  // 设置sh->len, sh->alloc
            SDS_HDR_VAR(8,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
        case SDS_TYPE_16: {
            SDS_HDR_VAR(16,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
    }
    if (initlen && init)    
        memcpy(s, init, initlen);
    s[initlen] = '\0';
    return s;
}

sds sdscpylen(sds s, const char *t, size_t len) {   // char* t拷贝给s, 调用memcpy
    if (sdsalloc(s) < len) {
        s = sdsMakeRoomFor(s,len-sdslen(s));
        if (s == NULL) return NULL;
    }
    memcpy(s, t, len);
    s[len] = '\0';
    sdssetlen(s, len);
    return s;
}
int sdscmp(const sds s1, const sds s2) {    // 比较, 调用memcmp
    size_t l1, l2, minlen;
    int cmp;

    l1 = sdslen(s1);
    l2 = sdslen(s2);
    minlen = (l1 < l2) ? l1 : l2;
    cmp = memcmp(s1,s2,minlen);
    if (cmp == 0) return l1>l2? 1: (l1<l2? -1: 0);
    return cmp;
}
```

### Dict 字典


redis字典其实就是基于hash的key-value map, 由小到大的为dictEntry字典元素->dictht哈希表(指向连续的字典元素)->字典对象(维护两个hash表)

dict对象维护两个hash表, 目的是hash表在运行时可能扩展或者收缩,。平时用的是table0, 当扩展或收缩一定大小时先给table1开辟对应大小空间, 再将table0元素复制到table1中, 最后将table1设置为table0, 原来的table0析构。

字典在Redis中的应用相当广泛, 比如Redis的数据库就是使用字典来作为底层实现, 对数据库的增，删，查，改都是构建在字典之上的。

dict.h
```cpp
typedef struct dictEntry {  // 字典维护的元素, key-value, 其中value由于可取不同类型设置是一个union
    void *key;
    union {
        void *val;
        uint64_t u64;
        int64_t s64;
        double d;
    } v;
    struct dictEntry *next;
} dictEntry;

typedef struct dictType {   // 字典排布entry需要实现的函数, hashfunction, 比较, 复制, 析构
    uint64_t (*hashFunction)(const void *key);
    void *(*keyDup)(void *privdata, const void *key);
    void *(*valDup)(void *privdata, const void *obj);
    int (*keyCompare)(void *privdata, const void *key1, const void *key2);
    void (*keyDestructor)(void *privdata, void *key);
    void (*valDestructor)(void *privdata, void *obj);
} dictType;

typedef struct dictht { // hash表, dictEntry的双重指针, 简单的一重指针是指向一个数组(hash bucket), 数组的元素也是指针指向Entry
    dictEntry **table;
    unsigned long size;
    unsigned long sizemask;
    unsigned long used;
} dictht;

typedef struct dict {   // 字典对象, 包括两个hash表, 迭代器, rehash作用是扩充收缩hash表, 简单的ht[1]设置新的大小将ht[0]复制到ht[1], 析构ht[0],最后将ht[1]设为ht[0]
    dictType *type;
    void *privdata;
    dictht ht[2];
    long rehashidx; /* rehashing not in progress if rehashidx == -1 */
    unsigned long iterators; /* number of iterators currently running */
} dict;

typedef struct dictIterator {   // hash表迭代器, 内部维护当前元素entry
    dict *d;
    long index;
    int table, safe;
    dictEntry *entry, *nextEntry;
    /* unsafe iterator fingerprint for misuse detection. */
    long long fingerprint;
} dictIterator;

// dict常用属性函数
#define dictHashKey(d, key) (d)->type->hashFunction(key)
#define dictGetKey(he) ((he)->key)
#define dictGetVal(he) ((he)->v.val)
#define dictSize(d) ((d)->ht[0].used+(d)->ht[1].used)
#define dictIsRehashing(d) ((d)->rehashidx != -1)   // rehashidx==-1说明不在rehash, rehashidx>=0说明rehash 哈希表的当前索引

/* API */
dict *dictCreate(dictType *type, void *privDataPtr);    // 刚创建的字典对象的hash表为0
int dictExpand(dict *d, unsigned long size);    // 执行了d->ht[1]的扩展, 包括创建
int dictReplace(dict *d, void *key, void *val);
int dictResize(dict *d);
dictIterator *dictGetIterator(dict *d);
dictIterator *dictGetSafeIterator(dict *d);
dictEntry *dictNext(dictIterator *iter);
int dictAdd(dict *d, void *key, void *val);
int dictRehash(dict *d, int n);
int dictRehashMilliseconds(dict *d, int ms);
```

Dict.c 实现
```cpp

// 创建Dict
static void _dictReset(dictht *ht)
{
    ht->table = NULL;   // 初始化hash table, 大小为0
    ht->size = 0;
    ht->sizemask = 0;
    ht->used = 0;
}

/* Create a new hash table */
dict *dictCreate(dictType *type,
        void *privDataPtr)
{
    dict *d = zmalloc(sizeof(*d));  // 创建字典对象

    _dictInit(d,type,privDataPtr);  // 初始化字典对象的元素
    return d;
}

/* Initialize the hash table */
int _dictInit(dict *d, dictType *type,
        void *privDataPtr)
{
    _dictReset(&d->ht[0]);  // 设置两个hash表
    _dictReset(&d->ht[1]);
    d->type = type;
    d->privdata = privDataPtr;
    d->rehashidx = -1;
    d->iterators = 0;
    return DICT_OK;
}

// 增加一个元素
int dictAdd(dict *d, void *key, void *val)
{
    dictEntry *entry = dictAddRaw(d,key,NULL);  // 根据key, 在hashtable中找到一个位置前插, 返回插入后的entry指针

    if (!entry) return DICT_ERR;
    dictSetVal(d, entry, val);  // 设置entry的value
    return DICT_OK;
}

dictEntry *dictAddRaw(dict *d, void *key, dictEntry **existing)
{
    long index;
    dictEntry *entry;
    dictht *ht;

    if (dictIsRehashing(d)) _dictRehashStep(d);

    /* Get the index of the new element, or -1 if
     * the element already exists. */
    if ((index = _dictKeyIndex(d, key, dictHashKey(d,key), existing)) == -1)   // 使用hashfunction得到index
        return NULL;

    /* Allocate the memory and store the new entry.
     * Insert the element in top, with the assumption that in a database
     * system it is more likely that recently added entries are accessed
     * more frequently. */
    ht = dictIsRehashing(d) ? &d->ht[1] : &d->ht[0];    // 处于rehashing应该使用d->ht[1]
    entry = zmalloc(sizeof(*entry));    // 创建entry
    entry->next = ht->table[index]; // entry前插法插到ht->table[index]中
    ht->table[index] = entry;
    ht->used++;

    /* Set the hash entry fields. */
    dictSetKey(d, entry, key);  // entry已经在dict中,设置key
    return entry;
}

// rehash
int dictRehash(dict *d, int n) {    // 执行rehash
    int empty_visits = n*10; /* Max number of empty buckets to visit. */
    if (!dictIsRehashing(d)) return 0;

    while(n-- && d->ht[0].used != 0) {
        dictEntry *de, *nextde;

        /* Note that rehashidx can't overflow as we are sure there are more
         * elements because ht[0].used != 0 */
        assert(d->ht[0].size > (unsigned long)d->rehashidx);
        while(d->ht[0].table[d->rehashidx] == NULL) {
            d->rehashidx++;
            if (--empty_visits == 0) return 1;
        }
        de = d->ht[0].table[d->rehashidx];  // d->rehashidx， rehash hash表的当前索引 
        /* Move all the keys in this bucket from the old to the new hash HT 把d->ht[0].table[h]链表中的所有entry设置到d->ht[1]中*/
        while(de) {
            uint64_t h;

            nextde = de->next;
            /* Get the index in the new hash table */
            h = dictHashKey(d, de->key) & d->ht[1].sizemask;    // 根据de->key得到hash index h
            de->next = d->ht[1].table[h];
            d->ht[1].table[h] = de;
            d->ht[0].used--;
            d->ht[1].used++;
            de = nextde;
        }
        d->ht[0].table[d->rehashidx] = NULL;    // d->ht[0]
        d->rehashidx++; // 更新当前rehash的索引
    }

    /* Check if we already rehashed the whole table... */
    if (d->ht[0].used == 0) {
        zfree(d->ht[0].table);
        d->ht[0] = d->ht[1];
        _dictReset(&d->ht[1]);
        d->rehashidx = -1;  // rehash完毕, 设置d->rehashidx=-1
        return 0;
    }

    /* More to rehash... */
    return 1;
}
```

#### rehash

redis的扩容类似unordered_map, redis dict的负载因子定义也是元素数量/哈希桶长度。

扩容触发, 1服务器目前没有在执行 BGSAVE 命令或者 BGREWRITEAOF 命令， 并且哈希表的负载因子大于等于 1; 或者2.服务器目前正在执行BGSAVE 命令或者 BGREWRITEAOF 命令， 并且哈希表的负载因子大于等于 5;

收缩触发: 当负载因子小于等于0.1时，会执行收缩操作。

redis 的dict含有两个hashtable, 扩展或收缩哈希表需要将`ht[0]`里面的所有键值对rehash到`ht[1]`里面, 这个rehash动作并不是一次性、集中式地完成的，而是分多次、渐进式地完成的。redis的rehash是一个阶段, 不是立刻完成的

哈希渐进式rehash的详细步骤：

1. 为ht[1]分配空间，让字典同时持有ht[0]和ht[1]两个哈希表。

2. 在字典中维持一个索引计数器变量rehashidx，并将它的指设置为0，表示rehash工作正式开始。

3. 在rehash进行期间，每次对字典执行添加、删除、查找或者更新操作时，程序除了执行指定的操作以外，还会顺带将ht[0]哈希表在rehashidx索引上的所有键值对rehash到ht[1]，当rehash工作完成之后，程序将rehashidx属性的值一。

4. 随着字典操作的不断执行，最终在某个时间点，ht[0]的所有键值对都会被rehash至ht[1]，这时程序将rehashidx属性设置为-1，表示rehash已经操作完成

例如以下函数也会顺带进行rehash, 添加过程中，会先判断一下还有没有有需要执行的rehash操作，如果有，那就顺带进行rehash操作。每次最多rehash10个元素。
```cpp
dictEntry *dictAddRaw(dict *d, void *key, dictEntry **existing)
{
    long index;
    dictEntry *entry;
    dictht *ht;
 
    if (dictIsRehashing(d)) _dictRehashStep(d);  //如果rehash操作未完成，进行一次rehash操作
 
    /* Get the index of the new element, or -1 if
     * the element already exists. */
    if ((index = _dictKeyIndex(d, key, dictHashKey(d,key), existing)) == -1)
        return NULL;
 
    /* Allocate the memory and store the new entry.
     * Insert the element in top, with the assumption that in a database
     * system it is more likely that recently added entries are accessed
     * more frequently. */
    ht = dictIsRehashing(d) ? &d->ht[1] : &d->ht[0];        // 顺带rehash
    entry = zmalloc(sizeof(*entry));   //为ht[1]开辟新的空间
    entry->next = ht->table[index];
    ht->table[index] = entry;
    ht->used++;
 
    /* Set the hash entry fields. */
    dictSetKey(d, entry, key);
    return entry;
}
```

### 整数集合

整数集合是一种用数组实现的集合, 其不包含重复元素且元素按照从小到大排序, 由于元素有序, 可以使用二分查找时间复杂度为`log2(n)`。

相比于红黑树, 整数集合占用空间少(不用保存left, right, parent这种节点), 查询时间不弱。但是插入时间慢O(n), 因此尽量不要存储集合元素太大。

注意整数集合声明`int8_t contents[]`, 但其并不储存`int8_t`类型的整数, 存储的整数类型取决于`encoding`,可以升级降级。初始类型是`INTSET_ENC_INT16`, 如果某个整数过大, 可以升级为`INTSET_ENC_INT32`,`INTSET_ENC_INT64`, 其实就是使用强制转换, 同样也可以退回去降级。

当一个集合只包含整数值元素，并且这个集合的元素数量不多时, Redis就会使用整数集合作为集合键的底层实现。

```cpp
// 创建空整数集合
/* Create an empty intset. */
intset *intsetNew(void) {
    intset *is = zmalloc(sizeof(intset));
    is->encoding = intrev32ifbe(INTSET_ENC_INT16);  // 编码是int16
    is->length = 0;
    return is;
}
// 二分查找
static uint8_t intsetSearch(intset *is, int64_t value, uint32_t *pos) {
    int min = 0, max = intrev32ifbe(is->length)-1, mid = -1;    // is集合从小到大排序的
    int64_t cur = -1;

    /* The value can never be found when the set is empty */
    if (intrev32ifbe(is->length) == 0) {
        if (pos) *pos = 0;
        return 0;
    } else {
        /* Check for the case where we know we cannot find the value,
         * but do know the insert position. */
        // 如果value比最大的还大, 或者比最小的还小
        if (value > _intsetGet(is,max)) {
            if (pos) *pos = intrev32ifbe(is->length);   // 设置位置为is->length
            return 0;
        } else if (value < _intsetGet(is,0)) {  // 设置value的位置为0
            if (pos) *pos = 0;
            return 0;
        }
    }
    while(max >= min) {
        mid = ((unsigned int)min + (unsigned int)max) >> 1;
        cur = _intsetGet(is,mid);
        // 二分查找
        if (value > cur) {
            min = mid+1;
        } else if (value < cur) {
            max = mid-1;
        } else {
            break;
        }
    }
    if (value == cur) { // 如果已经存在于集合中
        if (pos) *pos = mid;
        return 1;
    } else {
        if (pos) *pos = min;    // value应该处于的新位置
        return 0;
    }
}

// 插入元素, 注意value这里类型虽然是int64_t, 但可以根据其实际值选择int64, int32, int16。
intset *intsetAdd(intset *is, int64_t value, uint8_t *success) {
    uint8_t valenc = _intsetValueEncoding(value);   // value的类型可以放在INTSET_ENC_INT64, INTSET_ENC_INT32, INTSET_ENC_INT16
    uint32_t pos;
    if (success) *success = 1;
    // value返回过大, 需要升级操作
    if (valenc > intrev32ifbe(is->encoding)) {  // 如果新元素类型比整数集合现有类型长, 整数集合需要升级才能添加新元素
        /* This always succeeds, so we don't need to curry *success. */
        return intsetUpgradeAndAdd(is,value);
    } else {
        // 直接二分查找要插入的位置
        if (intsetSearch(is,value,&pos)) {  // pos返回value添加的位置
            if (success) *success = 0;
            return is;
        }

        is = intsetResize(is,intrev32ifbe(is->length)+1);   // 重新设置size = length+1
        if (pos < intrev32ifbe(is->length)) intsetMoveTail(is,pos,pos+1);   // 将pos和以后的元素向右移动一个位置
    }

    _intsetSet(is,pos,value);   // 设置pos为value的位置
    is->length = intrev32ifbe(intrev32ifbe(is->length)+1);
    return is;
}

// 升级类型, 注意对于需要升级的数插入是十分容易的。对于正数, 不够放说明value太大了。对于负数说明value小, 因此不是插第一个就是最后一个
static intset *intsetUpgradeAndAdd(intset *is, int64_t value) { 
    uint8_t curenc = intrev32ifbe(is->encoding);
    uint8_t newenc = _intsetValueEncoding(value);   // value的enc类型
    int length = intrev32ifbe(is->length);
    int prepend = value < 0 ? 1 : 0;    

    /* First set new encoding and resize */
    is->encoding = intrev32ifbe(newenc);    // 设置is新的编码类型
    is = intsetResize(is,intrev32ifbe(is->length)+1);   // 扩展集合数组的大小

    while(length--)
        _intsetSet(is,length+prepend,_intsetGetEncoded(is,length,curenc));  // 这里将集合is的元素重新赋值为新的类型intrev32ifbe(newenc)的元素

    /* Set the value at the beginning or the end. */
    if (prepend)
        _intsetSet(is,0,value);
    else
        _intsetSet(is,intrev32ifbe(is->length),value);  // 对于正数, 不够放说明value太大了, 因此value直接放到集合最后一个
    is->length = intrev32ifbe(intrev32ifbe(is->length)+1);
    return is;
}

/* Set the value at pos, using the configured encoding. */ // 在pos处设置新的value, 升级强制转换
static void _intsetSet(intset *is, int pos, int64_t value) {
    uint32_t encoding = intrev32ifbe(is->encoding);

    if (encoding == INTSET_ENC_INT64) {
        ((int64_t*)is->contents)[pos] = value;  // 升级强制转换is->contents
        memrev64ifbe(((int64_t*)is->contents)+pos);
    } else if (encoding == INTSET_ENC_INT32) {
        ((int32_t*)is->contents)[pos] = value;
        memrev32ifbe(((int32_t*)is->contents)+pos);
    } else {
        ((int16_t*)is->contents)[pos] = value;
        memrev16ifbe(((int16_t*)is->contents)+pos);
    }
}
```

### 跳跃表

跳跃表的实现的server.h和t_zet.c中。

每个跳跃表节点存储所有level的前向节点, 和level0的后向节点。跳跃表的header节点不存储数据。

level从0开始, 每个Level的跳跃表节点都是从小到大排序的, 而某个节点能伸展多少个level, 由随机决定。

在Redis中,跳跃表的作用是取代红黑树, 作为有序集合键的实现之一, 即set。

server.h中跳跃表的定义

t_zset.c的跳跃表创建插入实现
```cpp
// 创建跳跃表, 实际只创建了header一个节点
zskiplist *zslCreate(void) {
    int j;
    zskiplist *zsl;
    // 分配内存, 只设置了一个节点zsl->header
    zsl = zmalloc(sizeof(*zsl));
    zsl->level = 1;
    zsl->length = 0;
    zsl->header = zslCreateNode(ZSKIPLIST_MAXLEVEL,0,NULL);
    for (j = 0; j < ZSKIPLIST_MAXLEVEL; j++) {
        zsl->header->level[j].forward = NULL;   // 先只有一层
        zsl->header->level[j].span = 0;
    }
    zsl->header->backward = NULL;
    zsl->tail = NULL;
    return zsl;
}

// 随机层数, 位于1~ZSKIPLIST_MAXLEVEL
int zslRandomLevel(void) {
    int level = 1;
    while ((random()&0xFFFF) < (ZSKIPLIST_P * 0xFFFF))
        level += 1;
    return (level<ZSKIPLIST_MAXLEVEL) ? level : ZSKIPLIST_MAXLEVEL;
}

// 插入节点
zskiplistNode *zslInsert(zskiplist *zsl, double score, sds ele) { 
    zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
    unsigned int rank[ZSKIPLIST_MAXLEVEL];
    int i, level;

    serverAssert(!isnan(score));
    x = zsl->header; // x先设置为头节点
    for (i = zsl->level-1; i >= 0; i--) {   // 从高层到底层
        /* store rank that is crossed to reach the insert position */
        rank[i] = i == (zsl->level-1) ? 0 : rank[i+1];  // 每遍历一层时重置rank[i]=0
        while (x->level[i].forward &&
                (x->level[i].forward->score < score ||
                    (x->level[i].forward->score == score &&
                    sdscmp(x->level[i].forward->ele,ele) < 0))) // 首先根据score排序, score相同则ele字典序
        {
            rank[i] += x->level[i].span;    // 从每个level header走到elem插入位置前一个节点的跨度(从0层算需要的节点数量)
            x = x->level[i].forward;    // x前进
        }
        update[i] = x;  //每个level需要插入的ele在x节点的后面, 记录下来
    }
    level = zslRandomLevel(); // 随机的层数(ele所属的)
    if (level > zsl->level) {   // 如果随机层数level大于现有跳跃表的层数, 首先需要在多出的层数设置
        for (i = zsl->level; i < level; i++) {
            rank[i] = 0;
            update[i] = zsl->header;    // 第i层需要插入的位置在header后面
            update[i]->level[i].span = zsl->length; // update节点到forward节点(就是end)距离为zsl->length
        }
        zsl->level = level;
    }
    x = zslCreateNode(level,score,ele); // 插入的节点创建
    for (i = 0; i < level; i++) {   // 从0到随机的level层
        x->level[i].forward = update[i]->level[i].forward;  // 将x在level[i]层插入到update和update->forward之间
        update[i]->level[i].forward = x;

        /* update span covered by update[i] as x is inserted here */
        x->level[i].span = update[i]->level[i].span - (rank[0] - rank[i]);  // 修改x和update节点的token值, 
        //update[i]->level[i].span表示update到Update.forward的跨度, rank[i]表示第i层从header到update节点的跨度, (rank[0] - rank[i])时update到x的跨度, 相减为x到forward的跨度
        update[i]->level[i].span = (rank[0] - rank[i]) + 1;
    }

    /* increment span for untouched levels */
    for (i = level; i < zsl->level; i++) {
        update[i]->level[i].span++;
    }

    x->backward = (update[0] == zsl->header) ? NULL : update[0];    // backward是对第0层而言的
    if (x->level[0].forward)    
        x->level[0].forward->backward = x;
    else// 如果forward为空, x是tail, 可见尾节点是存储数据的
        zsl->tail = x;
    zsl->length++;  // 插一个节点, length++
    return x;
}
```

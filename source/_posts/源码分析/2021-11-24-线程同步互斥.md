### thread

C++11的thread代码主要位于`include/std/thread`中

thread一旦创建立即执行, 且必须对thread执行join或者detach。没啥, 就创建进程, join, detach, joinable而已。由于实现来自pthread库, 所以需要链接pthread。

```cpp
 /// thread
  class thread
  {
  public:
    typedef __gthread_t			native_handle_type;
    struct _Impl_base;
    typedef shared_ptr<_Impl_base>	__shared_base_type;

    /// thread::id 对象
    class id
    {
      native_handle_type	_M_thread;
    public:
      id() noexcept : _M_thread() { } //id
      explicit
      id(native_handle_type __id) : _M_thread(__id) { }

    private:
      friend class thread;
      friend class hash<thread::id>;

      friend bool
      operator==(thread::id __x, thread::id __y) noexcept
      { return __gthread_equal(__x._M_thread, __y._M_thread); }
    };

    // Simple base type that the templatized, derived class containing
    // an arbitrary functor can be converted to and called.
    struct _Impl_base
    {
      __shared_base_type	_M_this_ptr;

      inline virtual ~_Impl_base();

      virtual void _M_run() = 0;
    };

    template<typename _Callable>
      struct _Impl : public _Impl_base
      {
	_Callable		_M_func;

	_Impl(_Callable&& __f) : _M_func(std::forward<_Callable>(__f))
	{ }
  // 调用_M_func函数
	void
	_M_run() { _M_func(); }
      };
  // 私有对象id
  private:
    id				_M_id;

  public:
    thread() noexcept = default;
    // 拷贝构造和拷贝赋值被delete
    thread(thread&) = delete;
    thread(const thread&) = delete;
    // 移动构造被允许
    thread(thread&& __t) noexcept
    { swap(__t); }

    // 创建thread, 常用此函数
    // 直接注册函数_Callable&&和参数
    // 直接调用_M_start_thread运行
    template<typename _Callable, typename... _Args>
      explicit 
      thread(_Callable&& __f, _Args&&... __args)
      {
        _M_start_thread(_M_make_routine(std::__bind_simple(
                std::forward<_Callable>(__f),
                std::forward<_Args>(__args)...)));  // 构造完线程就开始执行了
      }

  // 线程对象析构
    ~thread()
    {
      if (joinable())
	std::terminate();
    }
  // 线程被join或者detach过, 即_M_id == id(), joinable()返回false
  // 如果调用join或者detach, 会手动设置_M_id = id()
    bool
    joinable() const noexcept
    { return !(_M_id == id()); }

    void
    join();

    void
    detach();

    thread::id
    get_id() const noexcept
    { return _M_id; }

    
    inline thread::id
    get_id() noexcept { return thread::id(__gthread_self()); }

    /// yield
    inline void
    yield() noexcept
    {
#ifdef _GLIBCXX_USE_SCHED_YIELD
      __gthread_yield();
#endif
    }
```

```cpp
// thread()创建后调用
  void
  thread::_M_start_thread(__shared_base_type __b)
  {
    if (!__gthread_active_p())
#if __EXCEPTIONS
      throw system_error(make_error_code(errc::operation_not_permitted),
			 "Enable multithreading to use std::thread");
#else
      __throw_system_error(int(errc::operation_not_permitted));
#endif

    __b->_M_this_ptr = __b;
    int __e = __gthread_create(&_M_id._M_thread,
			       &execute_native_thread_routine, __b.get());
    if (__e)
    {
      __b->_M_this_ptr.reset();
      __throw_system_error(__e);
    }
  }

  void
  thread::detach()
  {
    int __e = EINVAL;

    if (_M_id != id())
      __e = __gthread_detach(_M_id._M_thread);

    if (__e)
      __throw_system_error(__e);

    _M_id = id();
  }
    void
  thread::join()
  {
    int __e = EINVAL;

    if (_M_id != id())
      __e = __gthread_join(_M_id._M_thread, 0);

    if (__e)
      __throw_system_error(__e);

    _M_id = id();
  }
```

### mutex

没啥东西, 就是lock(), unlock()获取Mutex和释放mutex。
```cpp
  class __mutex_base
  {
  protected:
    typedef __gthread_mutex_t			__native_type;
    __native_type  _M_mutex = __GTHREAD_MUTEX_INIT; // 维护了_M_mutex
    constexpr __mutex_base() noexcept = default;


    __mutex_base() noexcept
    {
      // XXX EAGAIN, ENOMEM, EPERM, EBUSY(may), EINVAL(may)
      __GTHREAD_MUTEX_INIT_FUNCTION(&_M_mutex);
    }

    ~__mutex_base() noexcept { __gthread_mutex_destroy(&_M_mutex); }
#endif
  // 不可拷贝赋值
    __mutex_base(const __mutex_base&) = delete;
    __mutex_base& operator=(const __mutex_base&) = delete;
  };

/// mutex
  class mutex : private __mutex_base
  {
  public:
    typedef __native_type* 			native_handle_type;

#ifdef __GTHREAD_MUTEX_INIT
    constexpr
#endif
    mutex() noexcept = default;
    ~mutex() = default;

    mutex(const mutex&) = delete;
    mutex& operator=(const mutex&) = delete;

    void
    lock()
    {
      int __e = __gthread_mutex_lock(&_M_mutex);  // 调用获得mutex对象, mutex.lock()

      // EINVAL, EAGAIN, EBUSY, EINVAL, EDEADLK(may)
      if (__e)
	__throw_system_error(__e);
    }

    bool  // 调用__gthread_mutex_trylock尝试上锁
    try_lock() noexcept
    {
      // XXX EINVAL, EAGAIN, EBUSY
      return !__gthread_mutex_trylock(&_M_mutex);
    }

    void
    unlock()  // 释放Mutex
    {
      // XXX EINVAL, EAGAIN, EPERM
      __gthread_mutex_unlock(&_M_mutex);
    }
    // 内部维护的&_M_mutex结构体
    native_handle_type
    native_handle()
    { return &_M_mutex; }
  };
```

```cpp
template<typename _Mutex>
    class lock_guard
    {
    public:
    // _Mutex对象
      typedef _Mutex mutex_type;
      /// 需要用mutex构造lock_guard， 构造等于上锁
      explicit lock_guard(mutex_type& __m) : _M_device(__m)
      { _M_device.lock(); }

      lock_guard(mutex_type& __m, adopt_lock_t) : _M_device(__m)
      { } // calling thread owns mutex

      ~lock_guard()
      { _M_device.unlock(); } // 析构解锁
    // 去除拷贝构造和拷贝赋值
      lock_guard(const lock_guard&) = delete;
      lock_guard& operator=(const lock_guard&) = delete;

    private:
      mutex_type&  _M_device;
    };
```

相比lock_guard, unique_lock实现了更多功能, 例如手动解锁, try_lock等, 且内部持有mutex对象的指针。
```cpp
  template<typename _Mutex>
    class unique_lock
    {
    public:
      typedef _Mutex mutex_type;
    // 用mutex_type构造unique_lock
      explicit unique_lock(mutex_type& __m)
      : _M_device(&__m), _M_owns(false)
      {
	lock();
	_M_owns = true;
      }

      unique_lock(mutex_type& __m, defer_lock_t) noexcept
      // _M_device持有mutex的指针
      : _M_device(&__m), _M_owns(false)
      { }

      unique_lock(mutex_type& __m, try_to_lock_t)
      : _M_device(&__m), _M_owns(_M_device->try_lock())
      { }

      template<typename _Clock, typename _Duration>
	unique_lock(mutex_type& __m,
		    const chrono::time_point<_Clock, _Duration>& __atime)
	: _M_device(&__m), _M_owns(_M_device->try_lock_until(__atime))
	{ }

    // 析构解锁
      ~unique_lock()
      {
	if (_M_owns)
	  unlock();
      }

      unique_lock(const unique_lock&) = delete;
      unique_lock& operator=(const unique_lock&) = delete;
    // 移动语义
      unique_lock(unique_lock&& __u) noexcept
      : _M_device(__u._M_device), _M_owns(__u._M_owns)
      {
	__u._M_device = 0;
	__u._M_owns = false;
      }

      unique_lock& operator=(unique_lock&& __u) noexcept
      {
	if(_M_owns)
	  unlock();

	unique_lock(std::move(__u)).swap(*this);

	__u._M_device = 0;
	__u._M_owns = false;

	return *this;
      }
    // 手动上锁
      void
      lock()
      {
	if (!_M_device)
	  __throw_system_error(int(errc::operation_not_permitted));
	else if (_M_owns)
	  __throw_system_error(int(errc::resource_deadlock_would_occur));
	else
	  {
	    _M_device->lock();
	    _M_owns = true;
	  }
      }

      bool
      try_lock()
      {
	if (!_M_device)
	  __throw_system_error(int(errc::operation_not_permitted));
	else if (_M_owns)
	  __throw_system_error(int(errc::resource_deadlock_would_occur));
	else
	  {
	    _M_owns = _M_device->try_lock();
	    return _M_owns;
	  }
      }
// 手动解锁

      void
      unlock()
      {
	if (!_M_owns)
	  __throw_system_error(int(errc::operation_not_permitted));
	else if (_M_device)
	  {
	    _M_device->unlock();
	    _M_owns = false;
	  }
      }
// release释放对mutex的维护, 返回裸mutex的指针

      mutex_type*
      release() noexcept
      {
	mutex_type* __ret = _M_device;
	_M_device = 0;
	_M_owns = false;
	return __ret;
      }
  // 是否拥有锁
      bool
      owns_lock() const noexcept
      { return _M_owns; }

  // 返回持有的mutex
      mutex_type*
      mutex() const noexcept
      { return _M_device; }

    private:
      mutex_type*	_M_device;  
      bool		_M_owns; // XXX use atomic_bool
    };
```

可重入锁表示同一个线程可以多次获得锁(可以对某个带锁函数执行递归操作), 不可重入锁表示一个线程只能获得一个锁。其他线程都不能同时获得锁。

recursive_mutex对象和mutex对象正常用lock_guard或者unique_lock处理。

### 条件变量condition

注意使用condition_variable需要传入unique_lock, 因此需要用unique_lock进行加锁。常用的函数就是wait, notify_all()

```cpp
void do_print_id(int id)
{
    std::unique_lock <std::mutex> lck(mtx);
    while (!ready) // 如果标志位不为 true, 则等待...
        cv.wait(lck); // 当前线程被阻塞, 当全局标志位变为 true 之后,
    // 线程被唤醒, 继续往下执行打印线程编号id.
    std::cout << "thread " << id << '\n';
}

void go()
{
    std::unique_lock <std::mutex> lck(mtx);
    ready = true; // 设置全局标志位为 true.
    cv.notify_all(); // 唤醒所有线程.
}
```

```cpp
  /// condition_variable
  class condition_variable
  {
    typedef chrono::system_clock	__clock_t;
    typedef __gthread_cond_t		__native_type;  // 内部维护的条件变量

    __native_type			_M_cond = __GTHREAD_COND_INIT;


  public:
    typedef __native_type* 		native_handle_type;

    condition_variable() noexcept;
    ~condition_variable() noexcept;
    // 禁止拷贝和赋值
    condition_variable(const condition_variable&) = delete;
    condition_variable& operator=(const condition_variable&) = delete;
    // 随机唤醒一个线程
    void
    notify_one() noexcept;
  // 唤醒所有线程
    void
    notify_all() noexcept;
  // 该线程进入阻塞, wait传入unique_lock类型
    
    void
    wait(unique_lock<mutex>& __lock);
  // _Predicate为true时跳出循环, 直接执行wait(uniquelock, is==2)即可
    template<typename _Predicate>
      void
      wait(unique_lock<mutex>& __lock, _Predicate __p)
      {
	while (!__p())
	  wait(__lock);
      }
  // 带有超时时间的wait
    template<typename _Duration>
      cv_status
      wait_until(unique_lock<mutex>& __lock,
		 const chrono::time_point<__clock_t, _Duration>& __atime)
      { return __wait_until_impl(__lock, __atime); }


    template<typename _Clock, typename _Duration, typename _Predicate>
      bool
      wait_until(unique_lock<mutex>& __lock,
		 const chrono::time_point<_Clock, _Duration>& __atime,
		 _Predicate __p)
      {
	while (!__p())
  // 先等待时间再判断条件
	  if (wait_until(__lock, __atime) == cv_status::timeout)
	    return __p();
	return true;
      }

    // wait_for, 等一段时间
    template<typename _Rep, typename _Period, typename _Predicate>
      bool
      wait_for(unique_lock<mutex>& __lock,
	       const chrono::duration<_Rep, _Period>& __rtime,
	       _Predicate __p)
      { return wait_until(__lock, __clock_t::now() + __rtime, std::move(__p)); }

  // 内部维护的condition变量
    native_handle_type
    native_handle()
    { return &_M_cond; }

  private:

  // 超时的等待, 调用__gthread_cond_timedwait
    template<typename _Dur>
      cv_status
      __wait_until_impl(unique_lock<mutex>& __lock,
			const chrono::time_point<__clock_t, _Dur>& __atime)
      {
	auto __s = chrono::time_point_cast<chrono::seconds>(__atime);
	auto __ns = chrono::duration_cast<chrono::nanoseconds>(__atime - __s);

	__gthread_time_t __ts =
	  {
	    static_cast<std::time_t>(__s.time_since_epoch().count()),
	    static_cast<long>(__ns.count())
	  };

	__gthread_cond_timedwait(&_M_cond, __lock.mutex()->native_handle(),
				 &__ts);

	return (__clock_t::now() < __atime
		? cv_status::no_timeout : cv_status::timeout);
      }
  };

// 相关实现, 构造

  condition_variable::condition_variable() noexcept
  {
    __GTHREAD_COND_INIT_FUNCTION(&_M_cond);
  }

  condition_variable::~condition_variable() noexcept
  {
    // XXX no thread blocked
    /* int __e = */ __gthread_cond_destroy(&_M_cond);
    // if __e == EBUSY then blocked
  }
// 调用__gthread_cond_wait
  void
  condition_variable::wait(unique_lock<mutex>& __lock)
  {

    if (__e)
      __throw_system_error(__e);
  }

  condition_variable::notify_one() noexcept
  {

  }

  {

  }
```

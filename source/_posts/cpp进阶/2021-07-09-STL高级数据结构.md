### 堆和优先队列

STL引用优先队列使用的头文件是`queue`。

```cpp
#### priority_queue优先队列

优先队列是一个有权值的队列。由于这是一个队列，所以只允许尾端加入元素，顶端取出元素。而内部元素排列并非按照推入的次序排列，而是自动按照权值排列。

优先队列底层用最大堆实现，堆元素的插入其实是上溯+下溯两个操作。先自底向上判断父节点元素是否大于子节点，若否，修改之并以子节点自顶向下修正。

```cpp
#include <queue>

template <class T, class Sequence = vector<T>,
class Compare = less<typename Sequence::value_type>>

class priority_queue {
protected:
    Sequence c; // 底层容器
    Compare comp;   // 元素大小比较标准
public:
    priority_queue() : c() {}
    explicit priority_queue(const Compare& x) : c(), comp(x) {}
    template <class InputIterator>
    priority_queue(InputIterator first, InputIterator last)
    : c (first, last) {
        make_heap(c.begin(), c.end(), comp);
    }
    bool empty() const {
        return c.empty();
    }
    size_type size() const {
        return c.size();
    }
    const_reference top() const {
        return c.front();
    }
    void push(const value_type& x) {
        c.push_back(x);
        // 重排heap
        push_heap(c.begin(), c.end(), comp);
    }
    void pop() {
        pop_heap(c.begin, c.end(), comp);
        c.pop_back();
    }
}

make_heap和push_heap, 位于`stl_heap.h`中
```cpp
  template<typename _RandomAccessIterator, typename _Compare>
    void
    make_heap(_RandomAccessIterator __first, _RandomAccessIterator __last,
	      _Compare __comp)  // 将一个数组构造成堆
    {
      typedef typename iterator_traits<_RandomAccessIterator>::value_type
	  _ValueType;
      typedef typename iterator_traits<_RandomAccessIterator>::difference_type
	  _DistanceType;

      if (__last - __first < 2)
	return;

      const _DistanceType __len = __last - __first;
      _DistanceType __parent = (__len - 2) / 2; // __parent是size/2处的节点, 最后一个元素的父亲
      while (true)
	{
	  _ValueType __value = _GLIBCXX_MOVE(*(__first + __parent));
	  std::__adjust_heap(__first, __parent, __len, _GLIBCXX_MOVE(__value),
			     __comp);   // 调整堆, 自上而下
	  if (__parent == 0)
	    return;
	  __parent--;   // 调整父亲， 自下而上
	}
    }


  template<typename _RandomAccessIterator, typename _Distance, typename _Tp>
  template<typename _RandomAccessIterator, typename _Distance,
	   typename _Tp, typename _Compare>
    void
    __adjust_heap(_RandomAccessIterator __first, _Distance __holeIndex,
		  _Distance __len, _Tp __value, _Compare __comp)    // 从当前节点向下的调整堆, 如果Compare=less<T>说明越小的下移动, 是大顶堆
    {
      const _Distance __topIndex = __holeIndex;
      _Distance __secondChild = __holeIndex;
      while (__secondChild < (__len - 1) / 2)
	{
	  __secondChild = 2 * (__secondChild + 1);
	  if (__comp(*(__first + __secondChild),
		     *(__first + (__secondChild - 1))))
	    __secondChild--;    // 子节点的比较, 默认less<T>, 即__secondChild<__secondChild-1返回true,使__secondChild--, 也就是说不选择less<T>返回true的。 这样反而返回较大的那个子节点
	  *(__first + __holeIndex) = _GLIBCXX_MOVE(*(__first + __secondChild)); // 移动赋值位置, std::move
	  __holeIndex = __secondChild;  // 继续向下
	}
      if ((__len & 1) == 0 && __secondChild == (__len - 2) / 2) 
	{
	  __secondChild = 2 * (__secondChild + 1);
	  *(__first + __holeIndex) = _GLIBCXX_MOVE(*(__first
						     + (__secondChild - 1)));
	  __holeIndex = __secondChild - 1;
	}
      std::__push_heap(__first, __holeIndex, __topIndex, 
		       _GLIBCXX_MOVE(__value), __comp);      // 如果是less<T>, 大的元素上移, 小的元素入堆。
    }

    #define _GLIBCXX_MOVE(__val) std::move(__val)



      template<typename _RandomAccessIterator, typename _Compare>
    inline void
    push_heap(_RandomAccessIterator __first, _RandomAccessIterator __last,
	      _Compare __comp)  // 元素加入堆， 先设置元素为最后一个位置，然后调整堆
    {
      typedef typename iterator_traits<_RandomAccessIterator>::value_type
	  _ValueType;
      typedef typename iterator_traits<_RandomAccessIterator>::difference_type
	  _DistanceType;

      _ValueType __value = _GLIBCXX_MOVE(*(__last - 1));
      std::__push_heap(__first, _DistanceType((__last - __first) - 1),
		       _DistanceType(0), _GLIBCXX_MOVE(__value), __comp);
    }

      template<typename _RandomAccessIterator, typename _Distance, typename _Tp,
	   typename _Compare>
    void
    __push_heap(_RandomAccessIterator __first, _Distance __holeIndex,
		_Distance __topIndex, _Tp __value, _Compare __comp) // 加入__value元素到堆
    {
      _Distance __parent = (__holeIndex - 1) / 2;   // 父元素
      while (__holeIndex > __topIndex
	     && __comp(*(__first + __parent), __value)) // 如果__comp是less<int>, 则parent<value将value移动到上面, 这是大顶堆
	{
	  *(__first + __holeIndex) = _GLIBCXX_MOVE(*(__first + __parent));
	  __holeIndex = __parent;   // 赋值parent为新位置
	  __parent = (__holeIndex - 1) / 2;
	}
      *(__first + __holeIndex) = _GLIBCXX_MOVE(__value);
    }

### 常用二分查找办法lower_bound, upper_bound

lower_bound, upper_bound来自`algorithm.h`头文件, 是STL支持一个一般二分查找办法。

一般的, 默认比较器是`less<int>`, 这样针对从小到大的序列，lower_bound可以找到大于等于_Tp的位置; upper_bound则找到大于_Tp的位置。

两个函数返回的是迭代器, 传入的也是迭代器。
```cpp 
  template<typename _ForwardIterator, typename _Tp, typename _Compare>
    _ForwardIterator
    lower_bound(_ForwardIterator __first, _ForwardIterator __last,
		const _Tp& __val, _Compare __comp) // lower_bound可以输入比较器
    {
      typedef typename iterator_traits<_ForwardIterator>::value_type
	_ValueType;
      typedef typename iterator_traits<_ForwardIterator>::difference_type
	_DistanceType;

      _DistanceType __len = std::distance(__first, __last);

      while (__len > 0)
	{
	  _DistanceType __half = __len >> 1;    // 二分
	  _ForwardIterator __middle = __first;
	  std::advance(__middle, __half);
	  if (__comp(*__middle, __val)) // 如果比较器是less<int>, middle<val时会缩小二分左边界__first=__middle
	    {
	      __first = __middle;
	      ++__first;
	      __len = __len - __half - 1;
	    }
	  else
	    __len = __half;
	}
      return __first;
    }

  template<typename _ForwardIterator, typename _Tp, typename _Compare>
    _ForwardIterator
    upper_bound(_ForwardIterator __first, _ForwardIterator __last,
		const _Tp& __val, _Compare __comp)
    {
      typedef typename iterator_traits<_ForwardIterator>::value_type
	_ValueType;
      typedef typename iterator_traits<_ForwardIterator>::difference_type
	_DistanceType;

      while (__len > 0)
	{
	  _DistanceType __half = __len >> 1;
	  _ForwardIterator __middle = __first;
	  std::advance(__middle, __half);
	  if (__comp(__val, *__middle))
	    __len = __half;
	  else
	    {
	      __first = __middle;   // 与lower_bound相反, 修改
	      ++__first;
	      __len = __len - __half - 1;
	    }
	}
      return __first;
    }

```

lower_bound, upper_bound返回的是迭代器, 一般使用
```cpp
vector<int>::iterator idx = upper_bound(times_.begin(), times_.end(), t);
return winpersons_[times_[idx-1-times_.begin()]];

//或者
int offset = upper_bound(times_.begin(), times_.end(), t) - -1-times_.begin();
return winpersons_[times_[offset]];
```

由于指针也是迭代器, 所以lower_bound也可以
```cpp
int num[6]={1,2,4,7,15,34}; 
sort(num,num+6);                           //按从小到大排序 
int pos1=lower_bound(num,num+6,7)-num;    // 输出首尾地址
int pos2=upper_bound(num,num+6,7)-num;    //
cout<<pos1<<" "<<num[pos1]<<endl;
cout<<pos2<<" "<<num[pos2]<<endl;
```

但是注意vector的迭代器可以加减运算, 但set不行。对于二叉树，红黑树我们可以调用upper_bound二分查找得到元素对应的迭代器，但不能进行迭代器的作差。因为vector的迭代器是随机访问迭代器(指针), 而tree只是前后向迭代器。

可以用advance给迭代器增加偏移量, 用distance求两个迭代器之间的距离。

### 仿函数

#### 仿函数

所谓的仿函数，就是基于`operator()`写法的class closure。

例如算术类仿函数
```cpp
template <class T>
struct plus : public binary_function<T> {
    T operator() (const T& x, const T& y) const {
        return x + y;
    }
};

template <class T>
struct modulus : public binary_function<T> {
    T operator() (const T& x, const T& y) const {
        return x % y;
    }
};

// 使用
plus<int> plusobj;
cout << plusobj(3,5) << endl;

accumulate(iv,begin(), iv.end(), plus<int>());

```

#### 配接器

所谓的配接器，功能类似于`std::bind`,而仿函数功能类似于`std::function<>`。配接器和仿函数的共同作为类似于`std::bind`与`std::function<>`的共同作用。这将一个函数或者成员函数表现成表达式的形式。

`not1(bind2nd(less<int>(), 12))`该表达式首先绑定`less<int>()`的参数为12，表示小于12。然后`not1`表示取反，意思是不小于12。

`count_if(iv.begin(), iv.end(), not1(bind2nd(less<int>(), 12)))`表示不小于12的元素个数。

```cpp
#include <iostream>
#include <vector>

class Shape
{ public: virtual void display() = 0; };

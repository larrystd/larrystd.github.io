> 多线程安全的数据结构


### 线程安全队列

队列在实践中通常用链表实现, 同时注意多线程条件下为了保证对象安全被销毁, 保证裸指针不会访问到被销毁的对象, 应该使用智能指针维护共享变量。

#### 有锁队列

```cpp
#include <memory>
#include <mutex>
#include <queue>
#include <condition_variable>

template<typename T>
class ThreadSafeQueue {
 private: 
    mutable std::mutex mut;
    std::queue<T> data_queue;
    std::condition_variable data_cond;

 public: 
    ThreadSafeQueue()
    {}

    void Push(T new_value) {
        std::lock_guard<std::mutex> lk(mut);
        data_queue.push(std::move(new_value));
        data_cond.notify_one();
    }

    void WaitAndPop(T& value) { // 传引用, 相当于共享变量
        std::unique_lock<std::mutex> lk(mut);
        data_cond.wait(lk, [this]{
            return !data_queue.empty();
        });
        value = std::move(data_queue.front());
        data_queue.pop();
    }

    std::shared_ptr<T> WaitAndPop() {
        std::unique_lock<std::mutex> lk(mut);
        data_cond.wait(lk, [this]{
            return !data_queue.empty();
        });

        std::shared_ptr<T> value = std::make_shared(::move(data_queue.front()));
        data_queue.pop();
        return value;
    }

    bool Empty() const {
        std::lock_guard<std::mutex> lk(mut);
        return data_queue.empty();
    }
};
```

用链表实现的有锁队列, 注意使用两个互斥器维护链表的头节点和尾节点
```cpp
template<typename T>
class ThreadSafeQueue{
 private:
    struct Node {
        std::shared_ptr<T> data;
        std::unique_ptr<Node> next;
    };
    std::mutex head_mutex;
    std::mutex tail_mutex;

    std::unique_ptr<Node> head_;    // head存储元素, tail不存储元素, 初始化head=tail=nullptr, 然后加一个元素tail就右移始终为nullptr
    Node* tail;

    Node* get_tail() {
        std::lock_guard<std::mutex> tail_lock(tail_mutex);
        return tail_;
    }

    std::unique_lock<Node> pop_head() {
        std::lock_guard<std::mutex> head_lock(head_mutex);
        if (head_.get() == get_tail) // 链表尚为空, 无法pop_head
            return nullptr;
        std::unique_ptr<Node> old_head = std::move(head_);
        head_ = std::move(old_head->next);
        return old_head;
    }
    
 public: 
    ThreadSafeQueue() : 
        head_(new Node), tail(head_.get())
    {}

    ThreadSafeQueue(const ThreadSafeQueue& other)=delete;
    ThreadSafeQueue& operator=(const ThreadSafeQueue& other)=delete;

    std::shared_ptr<T> try_pop() {
        std::unique_lock<T> new_data(std::make_shared<T>(std::move(new_value)));
        return old_head ? old_head->data:shared_ptr();
    }

    void push(T new_value) {    // 队尾添加, 注意tail是不存储元素的
        std::shared_ptr<T> new_data(std::make_shared(std::move(new_value)));
        std::unique_lock<Node> p(new Node);
        Node* const new_tail = p.get();
        std::lock_guard<std::mutex> tail_lock(tail_mutex);
        tail->data = new_data;
        tail->next = std::move(p);
        tail = new_tail;
    }
};
```

#### 无锁队列

最简单的无锁队列是, 用原子类维护链表的头节点和尾节点, 从而使队列的入队, 出队线程安全。这种方法可以保证一个线程执行入队, 一个线程执行出队的操作使安全的, 顾被称为single-producer, single-consumer。但如果多线程执行入队操作, 使用原子类维护尾节点不能够保证增加节点时的线程安全。

1. 用原子类维护无锁队列链表的head_和tail, 链表节点存储有数据, 用shared_ptr维护

```cpp
// single-producer, single consumer lock-free queue
template<typename T>
class LockfreeQueue {
 private: 
    struct Node {
        std::shared_ptr<T> data;
        Node* next;
        Node() 
            : next(nullptr)
        {}  // 注意delete Node不会delete掉data和next指向的数据, 因为它们是指针
    };

    std::atomic<Node*> head_;
    std::atomic<Node*> tail_;

    Node* popHead() {
        Node* const old_head = head_.load();    // 获取旧节点
        if (old_head == tail_.load()) {
            return nullptr;
        }
        head_.store(old_head->next);
        return old_head;
    }

 public: 
    LockfreeQueue():
        head_(new Node), tail_(head_.load())
    {}
    LockfreeQueue(const LockfreeQueue&) = delete;
    LockfreeQueue& operator=(const LockfreeQueue&) = delete;

    ~LockfreeQueue() {
        while(Node* const old_head = head_.load()) {
            head_.store(old_head->next);
            delete old_head;
        }
    }

    std::shared_ptr<T> Pop() {
        Node* old_head = popHead();
        if (!old_head) 
            return std::shared_ptr<T>();
        std::shared_ptr<T> const res(old_head->data);
        delete old_head;
        return res; // 返回节点中的数据
    }

    void push(T new_value) {
        std::shared_ptr<T> new_data(std::make_shared(new_value));
        Node* p = new Node;
        Node* const old_tail = tail_.load();
        old_tail->data.swap(new_data);
        old_tail->next = p; // tail节点不存储数据, 因此把new_value放在old_tail的位置上
        tail_.store(p);
    }
};
```

使用`bool __sync_bool_compare_and_swap (type * ptr, type oldval, type newval, …)`两个函数提供原子的比较和交换，如果*ptr == oldval,就将newval写入ptr，返回true。如果不相等则返回false。

```cpp
#pragma once

#include <atomic>
#include <string>
#include <iostream>
#include <list>
#include <stdio.h>

template <typename T>
class LinkList
{
    public:
        T data;
        LinkList<T> *next;
};

template <typename T>
class LockFreeQueue
{
    public:
        LockFreeQueue();
        //入队
        void push_back(T t);
        //出队
        T pop_front(void);
        //判队空
        bool isEmpty(void);
    private:
        LinkList<T> *head_;
        LinkList<T> *tail_;
        std::atomic_int elementNumbers_;       
};

template <typename T>
LockFreeQueue<T>::LockFreeQueue()
    :head_(NULL),
    tail_(new LinkList<T>),
    elementNumbers_(0)
{
    head_ = tail_;
    tail_->next = NULL;     
}

template <typename T>
void LockFreeQueue<T>::push_back(T t)
{
    auto newVal = new LinkList<T>;
    newVal->data = t;
    newVal->next = NULL;

    LinkList<T> *p;
    do
    {
        //此操作暂时可删除，之后会跟新
        p = tail_;
    }while(!__sync_bool_compare_and_swap(&tail_->next,NULL,newVal));

    //移动tail_
    __sync_bool_compare_and_swap(&tail_,tail_,newVal);
    elementNumbers_++;
}

template <typename T>
T LockFreeQueue<T>::pop_front()
{
    LinkList<T> *p;
    
    do
    {
        //获取第一个节点快照
        p = head_->next;
        if(!p)
        {
            return 0;    
        }
    }while(!__sync_bool_compare_and_swap(&head_->next,p,p->next));  // 直到head_->next == p, 则设置head->next=p->next
    elementNumbers_--;
    return p->data;   
}

template <typename T>
bool LockFreeQueue<T>::isEmpty()
{
    if(elementNumbers_ == 0)
    {
        return true;
    }
    else
    {
        return false;
    }
}
```

### STL线程安全

STL的容器不是线程安全的, 基于组合的模式可以实现线程安全的容器, 多线程使用时可以简单加锁封装。例如map, 注意获取stl的属性例如size()也需要加锁, 因为可能有线程再写。
```cpp
#ifndef UTILS_MAP_HPP__
#define UTILS_MAP_HPP__

#include <map>
#include <memory>
#include <mutex>

namespace utils
{
//thread safe map, need to free data youself, Key和Value是泛型
template<typename TKey, typename TValue>
class map
{
public:
	map() 
	{
	}
	virtual ~map() 
	{ 
		std::lock_guard<std::mutex> locker(m_mutexMap);
		m_map.clear(); 
	}
    // 多线程安全的插入元素
	bool insert(const TKey &key, const TValue &value, bool cover = false)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end() && cover)   // 如果找到了key, 先删除
		{
			m_map.erase(find);
		}

		auto result = m_map.insert(std::pair<TKey, TValue>(key, value));    // map insert默认是不覆盖, 而operaor[]会覆盖
		return result.second;
	}

    // 线程安全的删除元素erase
	void remove(const TKey &key)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end())
		{
			m_map.erase(find);
		}
	}
    // 线程安全查询某个元素
	bool lookup(const TKey &key, TValue &value)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end())
		{
			value = (*find).second;
			return true;
		}
		else
		{
			return false;
		}
	}

    // 获取size
	int size()
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);
		return m_map.size();
	}

public:
	std::mutex m_mutexMap;
	std::map<TKey, TValue> m_map;   // 实现了std::mapclass

}

#endif // UTILS_MAP_HPP__
```

### 线程池

1. 线程池内部需要有vector<thread>线程列表和queue<Task>任务队列
2. 调用线程池的start()会初始化线程池, 即启动n个线程执行任务并加入线程列表。线程执行的任务是从任务队列中获取任务并执行。起初线程会阻塞在等待任务队列添加任务

3. 有两个锁, 一个是处理任务队列, 一个用来处理获取线程池对象的单例

```cpp
typedef std::function<void()> Task;

class CMyThreadPool
{
private:
    int max_thread;                // max thread;
    int max_task;                // max task;
    // thread array:
    vector<thread> threads;
    // task queue:
    queue<Task> m_tasks;
    // lock:
    mutex m_lock;
    // condition:条件变量
    condition_variable has_task;
    bool running_flag;
public:
    ~CMyThreadPool(void);
　　//获取线程池对象指针
     static CMyThreadPool * getInstance();
　　//添加任务，成功返回true,失败返回false
     bool start(Task fun);
private:
    CMyThreadPool(void);
    bool InitThread();
    void DestroyPool();
　　//工作线程
    void WorkFun();
    static CMyThreadPool * m_pool;  // 指向单例的线程池对象
    static std::mutex *singal_mutex;
};


#include "MyThreadPool.h"

CMyThreadPool * CMyThreadPool::m_pool = NULL;
mutex* CMyThreadPool::singal_mutex = new mutex();

CMyThreadPool::CMyThreadPool(void):max_thread(default_max_thread),
    max_task(default_max_task),running_flag(true)
{
}


CMyThreadPool::~CMyThreadPool(void)
{
    DestroyPool();
}

CMyThreadPool * CMyThreadPool::getInstance()
{
    if( NULL == m_pool){
        //lock();
        std::lock_guard<std::mutex> l(*singal_mutex);
        if( NULL == m_pool){
            m_pool = new CMyThreadPool();
        }
        //unlock();
    }
    return m_pool;
}

bool CMyThreadPool::start( Task fun )
{
    //判断是否第一次，延缓线程初始化
    {
        if( threads.size() == 0){
            unique_lock<mutex> l(m_lock);
            if( threads.size() == 0){
                //初始化线程
                if(!InitThread()){
                    return false;
                }
            }
        }
    }
    //判断工作队列是否已满,没满则加入工作队列
    {
        unique_lock<mutex> l(m_lock);
        if( (unsigned int)max_task > m_tasks.size()){
            m_tasks.push(fun);  // 将fun添加到任务队列
        }else{
            return false;
        }
    }
    //唤醒一个线程
    has_task.notify_one();
    return true;
}


//已经上着锁了
bool CMyThreadPool::InitThread()
{
    for (int i = 0; i != max_thread; i++){  // 开启线程, 执行WorkFun, 加入线程列表
        threads.push_back(thread(&CMyThreadPool::WorkFun, this));
    }
    return true;
}

void CMyThreadPool::WorkFun()
{
    while(running_flag || !m_tasks.empty()){
        Task t;
        //获取task
        {
            unique_lock<mutex> l(m_lock);
            while( m_tasks.empty())
                has_task.wait(l);   // 等待任务队列
            t = m_tasks.front();
            m_tasks.pop();
        }
        //执行task
        t();
    }
}

void CMyThreadPool::DestroyPool()
{
    {
        unique_lock<mutex> u_lock(m_lock);
        running_flag = false;
    }
    has_task.notify_all();

    for( auto &t : threads){
        t.join();
    }
    threads.clear();
}
```

### 内存池

经典内存池有一个链表组成, 链表的每个节点是一个数据块(可以用char[]表示)。this->freeNodeHeader是一个数组, 可以直接访问到链表的节点

分配空间从freeNodeHeader开始返回一个FreeNode即可, 而释放一块空间直接从freeNodeHeader插入链表

内存池分配的空间在对象的operator new中可以直接设置使用

```cpp
#include <iostream>
using namespace std;

template<int ObjectSize, int NumofObjects = 20>   
class MemPool {
private:
    struct FreeNode {
        FreeNode* pNext;
        char data[ObjectSize];  // 使用数组在new FreeNode, data[ObjectSize]整体作为成员对象
    };

    FreeNode* freeNodeHeader;     //表示空闲链表
public:
    MemPool() {
        //init freeNodeHeader
        this->freeNodeHeader = new FreeNode[NumofObjects];  // 内存池的内存块大体是连着的
        // 设置FreeNode的链表相连
        for (int i = 0; i < NumofObjects - 1; i++) {
            freeNodeHeader[i].pNext = &freeNodeHeader[i + 1];
        }
        freeNodeHeader[NumofObjects-1].pNext = NULL;

    }

    ~MemPool() {
        cout << " ~ 析构函数 " << endl;
        delete[] freeNodeHeader;
    }

    //函数的实现在下面
    void* malloc() {
        if (freeNodeHeader==NULL) {  //空间没有了
            //分配一段新的空间
            cout << "重新添加了空间" << endl;
            FreeNode * newfreeNodeHeader = new FreeNode[NumofObjects];
            // 用链表串起来
            for (int i = 0; i < NumofObjects - 1; i++) {
                newfreeNodeHeader[i].pNext = &newfreeNodeHeader[i + 1];
            }
            newfreeNodeHeader[NumofObjects-1].pNext = NULL;   
            
            //于是又有了空间
            freeNodeHeader = newfreeNodeHeader;
        }
        cout << "分配一个空间" << endl;
        FreeNode * ret = freeNodeHeader;
        freeNodeHeader = freeNodeHeader->pNext;
        ret->pNext = NULL;
        return ret; // 返回一块FreeNode
    }

    void free(void* p) {
        cout << "释放一个空间" << endl;
        FreeNode* pNode = (FreeNode*)p;
        pNode->pNext = freeNodeHeader; //将释放的节点插入空闲节点头部
        freeNodeHeader = pNode;
    }
};
```

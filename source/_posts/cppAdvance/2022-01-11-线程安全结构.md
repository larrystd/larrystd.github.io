> 多线程安全的数据结构


### 线程安全队列

队列在实践中通常用链表实现, 同时注意多线程条件下为了保证对象安全被销毁, 保证裸指针不会访问到被销毁的对象, 应该使用智能指针维护共享变量。

#### 有锁队列

```cpp
#include <memory>
#include <mutex>
#include <queue>
#include <condition_variable>

template<typename T>
class ThreadSafeQueue {
 private: 
    mutable std::mutex mut;
    std::queue<T> data_queue;
    std::condition_variable data_cond;

 public: 
    ThreadSafeQueue()
    {}

    void Push(T new_value) {
        std::lock_guard<std::mutex> lk(mut);
        data_queue.push(std::move(new_value));
        data_cond.notify_one();
    }

    void WaitAndPop(T& value) { // 传引用, 相当于共享变量
        std::unique_lock<std::mutex> lk(mut);
        data_cond.wait(lk, [this]{
            return !data_queue.empty();
        });
        value = std::move(data_queue.front());
        data_queue.pop();
    }

    std::shared_ptr<T> WaitAndPop() {
        std::unique_lock<std::mutex> lk(mut);
        data_cond.wait(lk, [this]{
            return !data_queue.empty();
        });

        std::shared_ptr<T> value = std::make_shared(::move(data_queue.front()));
        data_queue.pop();
        return value;
    }

    bool Empty() const {
        std::lock_guard<std::mutex> lk(mut);
        return data_queue.empty();
    }
};
```

用链表实现的有锁队列, 注意使用两个互斥器维护链表的头节点和尾节点
```cpp
template<typename T>
class ThreadSafeQueue{
 private:
    struct Node {
        std::shared_ptr<T> data;
        std::unique_ptr<Node> next;
    };
    std::mutex head_mutex;
    std::mutex tail_mutex;

    std::unique_ptr<Node> head_;    // head存储元素, tail不存储元素, 初始化head=tail=nullptr, 然后加一个元素tail就右移始终为nullptr
    Node* tail;

    Node* get_tail() {
        std::lock_guard<std::mutex> tail_lock(tail_mutex);
        return tail_;
    }

    std::unique_lock<Node> pop_head() {
        std::lock_guard<std::mutex> head_lock(head_mutex);
        if (head_.get() == get_tail) // 链表尚为空, 无法pop_head
            return nullptr;
        std::unique_ptr<Node> old_head = std::move(head_);
        head_ = std::move(old_head->next);
        return old_head;
    }
    
 public: 
    ThreadSafeQueue() : 
        head_(new Node), tail(head_.get())
    {}

    ThreadSafeQueue(const ThreadSafeQueue& other)=delete;
    ThreadSafeQueue& operator=(const ThreadSafeQueue& other)=delete;

    std::shared_ptr<T> try_pop() {
        std::unique_lock<T> new_data(std::make_shared<T>(std::move(new_value)));
        return old_head ? old_head->data:shared_ptr();
    }

    void push(T new_value) {    // 队尾添加, 注意tail是不存储元素的
        std::shared_ptr<T> new_data(std::make_shared(std::move(new_value)));
        std::unique_lock<Node> p(new Node);
        Node* const new_tail = p.get();
        std::lock_guard<std::mutex> tail_lock(tail_mutex);
        tail->data = new_data;
        tail->next = std::move(p);
        tail = new_tail;
    }
};
```

#### 无锁队列

最简单的无锁队列是, 用原子类维护链表的头节点和尾节点, 从而使队列的入队, 出队线程安全。这种方法可以保证一个线程执行入队, 一个线程执行出队的操作使安全的, 顾被称为single-producer, single-consumer。但如果多线程执行入队操作, 使用原子类维护尾节点不能够保证增加节点时的线程安全。

```cpp
// single-producer, single consumer lock-free queue
template<typename T>
class LockfreeQueue {
 private: 
    struct Node {
        std::shared_ptr<T> data;
        Node* next;
        Node() 
            : next(nullptr)
        {}
    };

    std::atomic<Node*> head_;
    std::atomic<Node*> tail_;

    Node* popHead() {
        Node* const old_head = head_.load();
        if (old_head == tail_.load()) {
            return nullptr;
        }
        head_.store(old_head->next);
        return old_head;
    }

 public: 
    LockfreeQueue():
        head_(new Node), tail_(head_.load())
    {}
    LockfreeQueue(const LockfreeQueue&) = delete;
    LockfreeQueue& operator=(const LockfreeQueue&) = delete;

    ~LockfreeQueue() {
        while(Node* const old_head = head_.load()) {
            head_.store(old_head->next);
            delete old_head;
        }
    }

    std::shared_ptr<T> Pop() {
        Node* old_head = popHead();
        if (!old_head) 
            return std::shared_ptr<T>();
        std::shared_ptr<T> const res(old_head->data);
        delete old_head;
        return res;
    }

    void push(T new_value) {
        std::shared_ptr<T> new_data(std::make_shared(new_value));
        Node* p = new Node;
        Node* const old_tail = tail_.load();
        old_tail->data.swap(new_data);
        old_tail->next = p;
        tail_.store(p);
    }
};
```

### STL线程安全

STL的容器不是线程安全的, 多线程使用时可以简单加锁封装。例如map, 注意获取stl的属性例如size()也需要加锁, 因为可能有线程再写。
```cpp
#ifndef UTILS_MAP_HPP__
#define UTILS_MAP_HPP__

#include <map>
#include <memory>
#include <mutex>

namespace utils
{
//thread safe map, need to free data youself, Key和Value是泛型
template<typename TKey, typename TValue>
class map
{
public:
	map() 
	{
	}
	virtual ~map() 
	{ 
		std::lock_guard<std::mutex> locker(m_mutexMap);
		m_map.clear(); 
	}
    // 多线程安全的插入元素
	bool insert(const TKey &key, const TValue &value, bool cover = false)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end() && cover)   // 如果找到了key, 先删除
		{
			m_map.erase(find);
		}

		auto result = m_map.insert(std::pair<TKey, TValue>(key, value));    // map insert默认是不覆盖, 而operaor[]会覆盖
		return result.second;
	}

    // 线程安全的删除元素erase
	void remove(const TKey &key)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end())
		{
			m_map.erase(find);
		}
	}
    // 线程安全查询某个元素
	bool lookup(const TKey &key, TValue &value)
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);

		auto find = m_map.find(key);
		if (find != m_map.end())
		{
			value = (*find).second;
			return true;
		}
		else
		{
			return false;
		}
	}

    // 获取size
	int size()
	{
		std::lock_guard<std::mutex> locker(m_mutexMap);
		return m_map.size();
	}

public:
	std::mutex m_mutexMap;
	std::map<TKey, TValue> m_map;
};

}

#endif // UTILS_MAP_HPP__
```

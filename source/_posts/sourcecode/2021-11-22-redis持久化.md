### redis DB

redis DB可以看成若干字典存储的键值对存储, 持久化内容包括键值对和其他一些描述信息。

```cpp
// server.h
typedef struct redisDb {    // redisDb 数据库对象, 含有若干个dict
    dict *dict;                 /* db维护一个字典, redis db即是数据库也是表The keyspace for this DB */
    dict *expires;              /* 过期的键dict Timeout of keys with a timeout set */
    dict *blocking_keys;        /* client保持得keyKeys with clients waiting for data (BLPOP)*/
    dict *ready_keys;           /* Blocked keys that received a PUSH */
    dict *watched_keys;         /* WATCHED keys for MULTI/EXEC CAS */
    int id;                     /* Database ID */
    long long avg_ttl;          /* Average TTL, just for stats */
    list *defrag_later;         /* List of key names to attempt to defrag one by one, gradually. */
} redisDb;
```

DB 内部的主要对象是dict字典, 包括键值对字典, 过期键字典。redis本身设计成一个缓存, 每个键会设置一个期限, 超出期限就会过期删除。持久化不会持久化过期的键。此外当内存不足时, redis会基于LRU, LFU等淘汰策略将键值对删除。

### redis rio

redis持久化必然要基于IO进行磁盘读写, 因此redis封装了rio来干这些事情。

```cpp
// redis操作数据库的io
struct _rio {
    size_t (*read)(struct _rio *, void *buf, size_t len);
    size_t (*write)(struct _rio *, const void *buf, size_t len);
    off_t (*tell)(struct _rio *);   // 返回向前位置相对于buf起始的偏移量
    int (*flush)(struct _rio *);

    /* The current checksum */
    uint64_t cksum; // 校验和

    /* Backend-specific vars. */
    union {
        /* In-memory buffer target. */
        struct {
            sds ptr;
            off_t pos;  // 在ptr上的偏移量
        } buffer;
        /* Stdio file pointer target. */
        struct {
            FILE *fp;   // 文件指针
            off_t buffered; // 文件流内部的缓冲区/* Bytes written since last fsync. */
            off_t autosync; /* fsync after 'autosync' bytes written. */
        } file; // 文件读取io
        /* Multiple FDs target (used to write to N sockets). */
        struct {
            int *fds;       /* File descriptors. */
            int *state;     /* Error state of each fd. 0 (if ok) or errno. */
            int numfds;
            off_t pos;
            sds buf;
        } fdset;    // fdset 多fd
    } io;
};  // rio struct


// IO读取r的内容到buf中, 调用rio->read
static inline size_t rioRead(rio *r, void *buf, size_t len) {
    while (len) {
        size_t bytes_to_read = (r->max_processing_chunk && r->max_processing_chunk < len) ? r->max_processing_chunk : len;
        if (r->read(r,buf,bytes_to_read) == 0)
            return 0;
        if (r->update_cksum) r->update_cksum(r,buf,bytes_to_read);
        buf = (char*)buf + bytes_to_read;
        len -= bytes_to_read;
        r->processed_bytes += bytes_to_read;
    }
    return 1;
}

// 获取当前流位置的偏移rio->tell
static inline off_t rioTell(rio *r) {
    return r->tell(r);
}

// 调用rio->flush
static inline int rioFlush(rio *r) {
    return r->flush(r);
}

static size_t rioFileWrite(rio *r, const void *buf, size_t len) {   // 将Buf内容写入文件流，刷入文件
    size_t retval;

    retval = fwrite(buf,len,1,r->io.file.fp);   // 将buf内容写入到io.file.fp文件指针流
    r->io.file.buffered += len; // io fd缓冲区的偏移量

    if (r->io.file.autosync &&
        r->io.file.buffered >= r->io.file.autosync)
    {
        fflush(r->io.file.fp);
        redis_fsync(fileno(r->io.file.fp));
        r->io.file.buffered = 0;
    }
    return retval;
}

static size_t rioBufferWrite(rio *r, const void *buf, size_t len) {
    r->io.buffer.ptr = sdscatlen(r->io.buffer.ptr,(char*)buf,len);  // 调用sdscatlen将buf内容追加写到io.buffer.ptr的后面
    r->io.buffer.pos += len;
    return 1;
}

/* Returns 1 or 0 for success/failure. */
static size_t rioBufferRead(rio *r, void *buf, size_t len) {
    if (sdslen(r->io.buffer.ptr)-r->io.buffer.pos < len)
        return 0; /* not enough buffer to return len bytes. */
    memcpy(buf,r->io.buffer.ptr+r->io.buffer.pos,len);  // 将r->io.buffer.ptr的内容写到缓冲区buf中
    r->io.buffer.pos += len;
    return 1;
}
```

### redis RDB

RDB持久化的方式就是将redis db的内容持久化到文件系统中。

RDB的核心函数时rdbsave和rdbload。rdbsave执行步骤主要有,
1. 创建持久化文件, 命名`temp-$pid.rdb`
2. 向文件写入`REDIS$version`版本信息
以下每个db都会存储一列
3. 保存db标志`RDB_OPCODE_SELECTDB`, 持久化过期时间等信息
4. 遍历db中dict迭代器, 持久化db dict的的type, **以string形式持久化dict的key, 以object的形式持久化dict的value**。
5. 最后持久化lua脚本, 间隔标志EOF, 校验和


RDB中的核心思路是Copy-on-Write，来保证在进行快照操作的这段时间，需要压缩写入磁盘上的数据在内存中不会发生变化。在正常的快照操作中，一方面Redis主进程会fork一个快照进程执行RDB操作，另一方面这段时间发生的写操作造成的数据变化会以副本的方式存放在一个新的内存区域，待快照操作结束后才会同步到原来的内存区域, 在此期间可以读取老数据。

```cpp
int rdbSave(char *filename, rdbSaveInfo *rsi) {
    char tmpfile[256];
    char cwd[MAXPATHLEN]; /* Current working dir path for error messages. */
    FILE *fp;
    rio rdb;
    int error = 0;

    snprintf(tmpfile,256,"temp-%d.rdb", (int) getpid());    // 设置持久化文件名, 格式temp-%d.rdb
    fp = fopen(tmpfile,"w");    // 打开某个文件

    rioInitWithFile(&rdb,fp);   // 初始化rdn, fp

    if (rdbSaveRio(&rdb,&error,RDB_SAVE_NONE,rsi) == C_ERR) {   // saveRio
        errno = error;
        goto werr;
    }

    /* Make sure data will not remain on the OS's output buffers */
    if (fflush(fp) == EOF) goto werr;
    if (fsync(fileno(fp)) == -1) goto werr;
    if (fclose(fp) == EOF) goto werr;

    serverLog(LL_NOTICE,"DB saved on disk");
    server.dirty = 0;
    server.lastsave = time(NULL);
    server.lastbgsave_status = C_OK;
    return C_OK;

int rdbSaveRio(rio *rdb, int *error, int flags, rdbSaveInfo *rsi) {
    dictIterator *di = NULL;
    dictEntry *de;
    char magic[10];
    int j;
    uint64_t cksum;
    size_t processed = 0;

    if (server.rdb_checksum)
        rdb->update_cksum = rioGenericUpdateChecksum;
    snprintf(magic,sizeof(magic),"REDIS%04d",RDB_VERSION);  // 保存redis的version "REDIS%04d"
    if (rdbWriteRaw(rdb,magic,9) == -1) goto werr;  // 将redis version写入rdb中
    if (rdbSaveInfoAuxFields(rdb,flags,rsi) == -1) goto werr;
    if (rdbSaveModulesAux(rdb, REDISMODULE_AUX_BEFORE_RDB) == -1) goto werr;

    for (j = 0; j < server.dbnum; j++) {    // 保存server的所有db数据
        redisDb *db = server.db+j;
        dict *d = db->dict;
        if (dictSize(d) == 0) continue;
        di = dictGetSafeIterator(d);    // di时db所在dict的迭代器

        /* Write the SELECT DB opcode */
        if (rdbSaveType(rdb,RDB_OPCODE_SELECTDB) == -1) goto werr;  // 保存SelectDB 标志 RDB_OPCODE_SELECTDB
        if (rdbSaveLen(rdb,j) == -1) goto werr;

        uint64_t db_size, expires_size;
        db_size = dictSize(db->dict);
        expires_size = dictSize(db->expires);
        if (rdbSaveType(rdb,RDB_OPCODE_RESIZEDB) == -1) goto werr;
        if (rdbSaveLen(rdb,db_size) == -1) goto werr;   // 保存db_size
        if (rdbSaveLen(rdb,expires_size) == -1) goto werr;  // 保存expires_size

        /* Iterate this DB writing every entry */
        while((de = dictNext(di)) != NULL) {    // 遍历dictNext, 得到key-value pair
            sds keystr = dictGetKey(de);
            robj key, *o = dictGetVal(de);
            long long expire;

            initStaticStringObject(key,keystr);
            expire = getExpire(db,&key);    // 过期的键值
            if (rdbSaveKeyValuePair(rdb,&key,o,expire) == -1) goto werr;    // 持久化pair value过期的键不会保存到rdb中, 过期是一种键值, 过期的键自动被删除
        }
        dictReleaseIterator(di);
        di = NULL; /* So that we don't release it again on error. */
    }

    if (rsi && dictSize(server.lua_scripts)) {  // 持久化lua脚本server.lua_scripts
        di = dictGetIterator(server.lua_scripts);
        while((de = dictNext(di)) != NULL) {
            robj *body = dictGetVal(de);
            if (rdbSaveAuxField(rdb,"lua",3,body->ptr,sdslen(body->ptr)) == -1)
                goto werr;
        }
        dictReleaseIterator(di);
        di = NULL; /* So that we don't release it again on error. */
    }

    if (rdbSaveModulesAux(rdb, REDISMODULE_AUX_AFTER_RDB) == -1) goto werr;

    /* EOF opcode */
    if (rdbSaveType(rdb,RDB_OPCODE_EOF) == -1) goto werr;   // 持久化标志EOF

    cksum = rdb->cksum; // 校验和持久化
    memrev64ifbe(&cksum);
    if (rioWrite(rdb,&cksum,8) == 0) goto werr;
    return C_OK;
}

int rdbSaveKeyValuePair(rio *rdb, robj *key, robj *val, long long expiretime) {
    int savelru = server.maxmemory_policy & MAXMEMORY_FLAG_LRU; // PRU和LFU是两种缓存, 前者移出最远使用的，后者移出使用频率最低的
    int savelfu = server.maxmemory_policy & MAXMEMORY_FLAG_LFU;

    /* Save the expire time */
    if (expiretime != -1) { // 保存过期时间
        if (rdbSaveType(rdb,RDB_OPCODE_EXPIRETIME_MS) == -1) return -1;
        if (rdbSaveMillisecondTime(rdb,expiretime) == -1) return -1;
    }

    /* Save the LRU info. */
    if (savelru) {
        uint64_t idletime = estimateObjectIdleTime(val);
        idletime /= 1000; /* Using seconds is enough and requires less space.*/
        if (rdbSaveType(rdb,RDB_OPCODE_IDLE) == -1) return -1;
        if (rdbSaveLen(rdb,idletime) == -1) return -1;
    }

    /* Save the LFU info. */
    if (savelfu) {
        uint8_t buf[1];
        buf[0] = LFUDecrAndReturn(val);
        if (rdbSaveType(rdb,RDB_OPCODE_FREQ) == -1) return -1;
        if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
    }

    /* Save type, key, value */ // 保存type, key, val到rdb指针
    if (rdbSaveObjectType(rdb,val) == -1) return -1;
    if (rdbSaveStringObject(rdb,key) == -1) return -1;
    if (rdbSaveObject(rdb,val,key) == -1) return -1;    // 保存val
    return 1;
}
```

```cpp
int prepareForShutdown(int flags) { // 关闭之前要创建rdb持久化
    int save = flags & SHUTDOWN_SAVE;
    int nosave = flags & SHUTDOWN_NOSAVE;

    if (server.aof_state != AOF_OFF) {
        /* Kill the AOF saving child as the AOF we already have may be longer
         * but contains the full dataset anyway. */
        if (server.aof_child_pid != -1) {
            /* If we have AOF enabled but haven't written the AOF yet, don't
             * shutdown or else the dataset will be lost. */
            if (server.aof_state == AOF_WAIT_REWRITE) {
                serverLog(LL_WARNING, "Writing initial AOF, can't exit.");
                return C_ERR;
            }
            serverLog(LL_WARNING,
                "There is a child rewriting the AOF. Killing it!");
            kill(server.aof_child_pid,SIGUSR1);
        }
        /* Append only file: flush buffers and fsync() the AOF at exit */
        serverLog(LL_NOTICE,"Calling fsync() on the AOF file.");
        flushAppendOnlyFile(1);
        redis_fsync(server.aof_fd);
    }

    /* Create a new RDB file before exiting. */
    if ((server.saveparamslen > 0 && !nosave) || save) {
        serverLog(LL_NOTICE,"Saving the final RDB snapshot before exiting.");
        /* Snapshotting. Perform a SYNC SAVE and exit */
        rdbSaveInfo rsi, *rsiptr;
        rsiptr = rdbPopulateSaveInfo(&rsi);
        if (rdbSave(server.rdb_filename,rsiptr) != C_OK) {  // rdb持久化
            serverLog(LL_WARNING,"Error trying to save the DB, can't exit.");
            return C_ERR;
        }
    }
    /* Close the listening sockets. Apparently this allows faster restarts. */
    closeListeningSockets(1);
    serverLog(LL_WARNING,"%s is now ready to exit, bye bye...",
        server.sentinel_mode ? "Sentinel" : "Redis");
    return C_OK;
}

// shutdownCommand中将执行prepareForShutdown
```

### AOF

Redis是写后日志，Redis先执行命令，把数据写入内存，然后才记录日志。日志里记录的是Redis收到的每一条命令，这些命令是以文本形式保存。PS: 大多数的数据库采用的是写前日志(WAL)，例如MySQL，通过写前日志和两阶段提交，实现数据和逻辑的一致性。

和RDB直接持久化db的内容不同, AOF开始时, server对执行的命令同时会存储到buf中, 并定期将其buf中的命令进行持久化。此外redis还会定期对命令重写, 即通过创建一个新的AOF文件来替代现有的AOF文件，新旧两个文件所保存的数据库状态是相同的，但新的AOF存储是精简后的命令, 例如将`push 1; push 2`改成`push 1,2`, 因此体积会较旧AOF文件小很多。AOF日志也是写后日志, 但可以降低redis宕机之后的数据损失。

AOF日志记录Redis的每个写命令，步骤分为，命令追加（append）、文件写入（write）和文件同步（sync）。 
1. 命令追加, 当AOF持久化功能打开了，服务器在执行完一个写命令之后，会以协议格式将被执行的写命令追加到服务器的 aof_buf 缓冲区。 
2. 文件写入和同步

```cpp

void flushAppendOnlyFile(int force) {   // 顺序刷serverAOF到磁盘, 调用aofWrite并进行一些错误处理
    ssize_t nwritten;
    int sync_in_progress = 0;
    mstime_t latency;


    latencyStartMonitor(latency);
    nwritten = aofWrite(server.aof_fd,server.aof_buf,sdslen(server.aof_buf));   // 将server.aof_buf数据写入server.aof_fd
    latencyEndMonitor(latency);

    if (nwritten != (ssize_t)sdslen(server.aof_buf)) {
        static time_t last_write_error_log = 0;
        int can_log = 0;

        /* Limit logging rate to 1 line per AOF_WRITE_LOG_ERROR_RATE seconds. */
        if ((server.unixtime - last_write_error_log) > AOF_WRITE_LOG_ERROR_RATE) {
            can_log = 1;
            last_write_error_log = server.unixtime;
        }

        /* Log the AOF write error and record the error code. */
        if (nwritten == -1) {
            if (can_log) {
                serverLog(LL_WARNING,"Error writing to the AOF file: %s",
                    strerror(errno));   // 写入日志中
                server.aof_last_write_errno = errno;
            }
        }
    }
    /* Re-use AOF buffer when it is small enough. The maximum comes from the
     * arena size of 4k minus some overhead (but is otherwise arbitrary). */
    if ((sdslen(server.aof_buf)+sdsavail(server.aof_buf)) < 4000) {
        sdsclear(server.aof_buf);
    } else {
        sdsfree(server.aof_buf);
        server.aof_buf = sdsempty();    // aof_buf设置为空
    }

ssize_t aofWrite(int fd, const char *buf, size_t len) {     // 直接将buf数据写入到fd中
    ssize_t nwritten = 0, totwritten = 0;

    while(len) {
        nwritten = write(fd, buf, len);

        if (nwritten < 0) {
            if (errno == EINTR) {
                continue;
            }
            return totwritten ? totwritten : -1;
        }

        len -= nwritten;
        buf += nwritten;
        totwritten += nwritten;
    }

    return totwritten;
}
```

```cpp
// 调用feedAppendOnlyFile将cmd置入buf中
void propagate(struct redisCommand *cmd, int dbid, robj **argv, int argc,
               int flags)
{
    if (server.aof_state != AOF_OFF && flags & PROPAGATE_AOF)
        feedAppendOnlyFile(cmd,dbid,argv,argc);
    if (flags & PROPAGATE_REPL)
        replicationFeedSlaves(server.slaves,dbid,argv,argc);
}

// 执行命令函数call会调用propagate
void call(client *c, int flags) {   // 执行client->cmd
    long long dirty;
    ustime_t start, duration;
    int client_old_flags = c->flags;
    struct redisCommand *real_cmd = c->cmd;

    server.fixed_time_expire++;
    /* Call the command. */
    dirty = server.dirty;
    updateCachedTime(0);
    start = server.ustime;

    c->cmd->proc(c);    // 执行命令, 结果储存在c->buf
    duration = ustime()-start;
    dirty = server.dirty-dirty;
    if (dirty < 0) dirty = 0;

    /* Propagate the command into the AOF and replication link */
    if (flags & CMD_CALL_PROPAGATE &&
        (c->flags & CLIENT_PREVENT_PROP) != CLIENT_PREVENT_PROP)
    {

        /* However prevent AOF / replication propagation if the command
         * implementations called preventCommandPropagation() or similar,
         * or if we don't have the call() flags to do so. */
        if (c->flags & CLIENT_PREVENT_REPL_PROP ||
            !(flags & CMD_CALL_PROPAGATE_REPL))
                propagate_flags &= ~PROPAGATE_REPL;
        if (c->flags & CLIENT_PREVENT_AOF_PROP ||
            !(flags & CMD_CALL_PROPAGATE_AOF))
                propagate_flags &= ~PROPAGATE_AOF;

        /* Call propagate() only if at least one of AOF / replication
         * propagation is needed. Note that modules commands handle replication
         * in an explicit way, so we never replicate them automatically. */
        if (propagate_flags != PROPAGATE_NONE && !(c->cmd->flags & CMD_MODULE))
            propagate(c->cmd,c->db->id,c->argv,c->argc,propagate_flags);    // 调用propagate将命令传入AOF/replication
    }  
}

// cmd相关内容传入server.aof_buf中
void feedAppendOnlyFile(struct redisCommand *cmd, int dictid, robj **argv, int argc) {  // 将命令追加到server.aof_buf中
    sds buf = sdsempty();
    robj *tmpargv[3];

    /* The DB this command was targeting is not the same as the last command
     * we appended. To issue a SELECT command is needed. */
    if (dictid != server.aof_selected_db) {
        char seldb[64];

        snprintf(seldb,sizeof(seldb),"%d",dictid);
        buf = sdscatprintf(buf,"*2\r\n$6\r\nSELECT\r\n$%lu\r\n%s\r\n",
            (unsigned long)strlen(seldb),seldb);
        server.aof_selected_db = dictid;
    }

    if (cmd->proc == expireCommand || cmd->proc == pexpireCommand ||
        cmd->proc == expireatCommand) {
        /* Translate EXPIRE/PEXPIRE/EXPIREAT into PEXPIREAT */
        buf = catAppendOnlyExpireAtCommand(buf,cmd,argv[1],argv[2]);
    }

    /* Append to the AOF buffer. This will be flushed on disk just before
     * of re-entering the event loop, so before the client will get a
     * positive reply about the operation performed. */
    if (server.aof_state == AOF_ON)
        server.aof_buf = sdscatlen(server.aof_buf,buf,sdslen(buf)); // 讲command由buf加入到aof_buf中
}
```

#### AOF的调用

之前讲过在eventloop循环之前调用beforeSleep来讲`client->buf`的数据发送给客户端, 除此之外, beforeSleep期间还会执行依次写aof的操作
```cpp
// server里面维护有
sds aof_buf;      /* AOF buffer, written before entering the event loop */ // aof模式开启, 执行一个命令就把命令写入到缓冲区
int aof_fd;       /* File descriptor of currently selected AOF file */
int aof_selected_db; /* Currently selected DB in AOF */ // 当前打开得数据库

// init server中会打开AOF文件
/* Open the AOF file if needed. */
if (server.aof_state == AOF_ON) {
    server.aof_fd = open(server.aof_filename,
                            O_WRONLY|O_APPEND|O_CREAT,0644);
    if (server.aof_fd == -1) {
        serverLog(LL_WARNING, "Can't open the append-only file: %s",
            strerror(errno));
        exit(1);
    }
}

void beforeSleep(struct aeEventLoop *eventLoop) {   // eventloop的before sleep函数, 包括将cilent->buf的数据send到客户端
    /* Try to process pending commands for clients that were just unblocked. */
    if (listLength(server.unblocked_clients))
        processUnblockedClients();

    /* Write the AOF buffer on disk */
    flushAppendOnlyFile(0); // 写AOF文件

    /* Handle writes with pending output buffers. */
    handleClientsWithPendingWrites();   // 将输出buffers的数据发送给客户端

    /* Before we are going to sleep, let the threads access the dataset by
     * releasing the GIL. Redis main thread will not touch anything at this
     * time. */
    if (moduleCount()) moduleReleaseGIL();
}

void aeMain(aeEventLoop *eventLoop) {   // 事件执行主函数
    eventLoop->stop = 0;
    while (!eventLoop->stop) {
        if (eventLoop->beforesleep != NULL)
            eventLoop->beforesleep(eventLoop);  // 这里调用beforesleep
        aeProcessEvents(eventLoop, AE_ALL_EVENTS|AE_CALL_AFTER_SLEEP);  // 调用aeProcessEvents
    }
}
```

之前讲过server的定时器, 事实上server只执行一种定时器且是每个多少秒就会执行依次的, 也就是`serverCron`。持久化AOF任务包含在定时器中, 因此也是周期执行的。

```cpp
// server定期执行的一些任务, 每隔100毫秒执行一次, 包含完成过期key处理、软件监控、更新一些统计数据、触发RDB持久化或AOF重写、客户端超时处理等
#define run_with_period(_ms_) if ((_ms_ <= 1000/server.hz) || !(server.cronloops%((_ms_)/(1000/server.hz))))

int serverCron(struct aeEventLoop *eventLoop, long long id, void *clientData) {
    int j;

    /* Software watchdog: deliver the SIGALRM that will reach the signal
     * handler if we don't return here fast enough. */
    if (server.watchdog_period) watchdogScheduleSignal(server.watchdog_period);

    /* Update the time cache. */
    updateCachedTime(1);


    unsigned long lruclock = getLRUClock();
    atomicSet(server.lruclock,lruclock);

    /* Record the max memory used since the server was started. */
    if (zmalloc_used_memory() > server.stat_peak_memory)
        server.stat_peak_memory = zmalloc_used_memory();

    /* Show some info about non-empty databases */
    run_with_period(5000) { // 5000ms定期执行依次日志
        for (j = 0; j < server.dbnum; j++) {
            long long size, used, vkeys;

            size = dictSlots(server.db[j].dict);
            used = dictSize(server.db[j].dict);
            vkeys = dictSize(server.db[j].expires);
            if (used || vkeys) {
                serverLog(LL_VERBOSE,"DB %d: %lld keys (%lld volatile) in %lld slots HT.",j,used,vkeys,size);
                /* dictPrintStats(server.dict); */
            }
        }
    }

    /* Show information about connected clients */
    if (!server.sentinel_mode) {
        run_with_period(5000) { // 定期打印clients
            serverLog(LL_VERBOSE,
                "%lu clients connected (%lu replicas), %zu bytes in use",
                listLength(server.clients)-listLength(server.slaves),
                listLength(server.slaves),
                zmalloc_used_memory());
        }
    }

    /* We need to do a few operations on clients asynchronously. */
    clientsCron();  // 执行clientsCron

    /* Handle background operations on Redis databases. */
    databasesCron();    // 执行databasesCron

    /* Start a scheduled AOF rewrite if this was requested by the user while
     * a BGSAVE was in progress. */
    if (server.rdb_child_pid == -1 && server.aof_child_pid == -1 &&
        server.aof_rewrite_scheduled)
    {
        rewriteAppendOnlyFileBackground();  // 定期执行AOF rewrite
    }


    /* AOF postponed flush: Try at every cron cycle if the slow fsync
     * completed. */
    if (server.aof_flush_postponed_start) flushAppendOnlyFile(0);   // 定期执行flushAppendOnlyFile

    run_with_period(1000) {
        if (server.aof_last_write_status == C_ERR)
            flushAppendOnlyFile(0);
    }

    /* Close clients that need to be closed asynchronous */
    freeClientsInAsyncFreeQueue();

    /* Clear the paused clients flag if needed. */
    clientsArePaused(); /* Don't check return value, just use the side effect.*/

    /* Replication cron function -- used to reconnect to master,
     * detect transfer failures, start background RDB transfers and so forth. */
    run_with_period(1000) replicationCron();

    server.cronloops++;
    return 1000/server.hz;
}


// serverCron在InitServer期间被注册到eventloop的定时器中
if (aeCreateTimeEvent(server.el, 1, serverCron, NULL, NULL) == AE_ERR) {    // 将serverCron注册到server的定时器中, serverCron被定期执行
    serverPanic("Can't create event loop timers.");
    exit(1);
}
```

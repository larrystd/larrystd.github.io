#### strcpy 拷贝字符串

```cpp
char* strcpy(char* destination, const char* source);
char * strncpy ( char * destination, const char * source, size_t num );
/* Copy SRC to DEST.  */
char *
strcpy (char *dest, const char *src)
{
  char c;
  char *s = (char *) src;
  /// 直接找dest和src的地址差
  const ptrdiff_t off = dest - s - 1;

  do
    {
      /// 根据一个s指针就可以实现copy, 借助地址差
      c = *s++;
      s[off] = c; /// 通过地址差, s[off]就是dest
    }
  while (c != '\0');  /// 字符串结尾是 \0

  return dest;
}
```

```cpp
    {
      c1 = (unsigned char) *s1++;
```

#### strpbrk 从s中找到accept中的字符, 返回第一个的地址, 

当s为字符时可以充当匹配, 如`strpbrk("a", "abc");`

```cpp
/* Find the first occurrence in S of any character in ACCEPT.  */
char *
STRPBRK (const char *s, const char *accept)
{
  while (*s != '\0')
    {
      const char *a = accept;
      while (*a != '\0')
	if (*a++ == *s)
	  return (char *) s;
      ++s;
    }

  return NULL;
}
```

```cpp
/* Find the last occurrence of C in S.  */
char *
STRRCHR (const char *s, int c)
{
  const char *found, *p;

  c = (unsigned char) c;

  /* Since strchr is fast, we use it rather than the obvious loop.  */

  if (c == '\0')
    return strchr (s, '\0');

  found = NULL;
  while ((p = strchr (s, c)) != NULL)
    {
      found = p;
      s = p + 1;
    }

  return (char *) found;
}
```

#### strtok 字符串分割 将分割符传入

当strtok()在参数s 的字符串中发现到参数delim 的分割字符时则会将该字符改为\0 字符。在第一次调用时，strtok()必需给予参数s 字符串，往后的调用则将参数s 设置成NULL。

strtok执行一次将匹配分割delim位置置\0, 并返回起始指针, 这时候起始指针所表示的字符段就是分割的字符段。例如`abc-=-def`执行一次strtok(s,"-"),形成`abc\0=-def`返回起始指针s, 这时候strlen(s)为3, printf(s)为`abc`。

```cpp
/* Parse S into tokens separated by characters in DELIM.
   If S is NULL, the last string strtok() was called with is
   used.  For example:
	char s[] = "-abc-=-def";
	x = strtok(s, "-");		// x = "abc"
	x = strtok(NULL, "-=");		// x = "def"
	x = strtok(NULL, "=");		// x = NULL
		// s = "abc\0=-def\0"
*/
char *
STRTOK (char *s, const char *delim)
{
  char *token;

  if (s == NULL)
    s = olds;

  /* Scan leading delimiters.  */
  ///  strspn 检索字符串 str1 中第一个不在字符串 str2 中出现的字符下标。
  s += strspn (s, delim);
  /// 下一次s就指向 不匹配的第一个下标
  if (*s == '\0')
    {
      olds = s;
      return NULL;
    }

  /* Find the end of the token.  */
  token = s;
  s = strpbrk (token, delim);   /// 第一个delim出现位置, 就算delim -m, 可以单独和m匹配
  if (s == NULL)
    /* This token finishes the string.  */
    olds = __rawmemchr (token, '\0');
  else
    {
      /* Terminate the token and make OLDS point past it.  */
      *s = '\0';
      olds = s + 1;
    }
  return token;
}
```

#### memcpy

该文件用宏定义定义了字节拷贝和字拷贝

```cpp
typedef unsigned char byte; // byte用unsigned char表示

/* Copy exactly NBYTES bytes from SRC_BP to DST_BP,
   without any assumptions about alignment of the pointers.  */
/// 前向拷贝, 也就是从src起始一个字节一个字节拷贝到dst中
#define BYTE_COPY_FWD(dst_bp, src_bp, nbytes)				      \
  do									      \
    {									      \
      size_t __nbytes = (nbytes);					      \
      while (__nbytes > 0)						      \
	{								      \
	  byte __x = ((byte *) src_bp)[0];				      \
	  src_bp += 1;							      \
	  __nbytes -= 1;						      \
	  ((byte *) dst_bp)[0] = __x;					      \
	  dst_bp += 1;							      \
	}								      \
    } while (0)   /// dowhile, 不满足条件也至少执行一次

/* Copy exactly NBYTES_TO_COPY bytes from SRC_END_PTR to DST_END_PTR,
   beginning at the bytes right before the pointers and continuing towards
   smaller addresses.  Don't assume anything about alignment of the
   pointers.  */
/// 后向拷贝
/// 也就是从src结尾处一个字节一个字节拷贝到dst中
/// 传入的src_ep, dst_ep应该是结尾地址
#define BYTE_COPY_BWD(dst_ep, src_ep, nbytes)				      \
  do									      \
    {									      \
      size_t __nbytes = (nbytes);					      \
      while (__nbytes > 0)						      \
	{								      \
	  byte __x;							      \
	  src_ep -= 1;							      \
	  __x = ((byte *) src_ep)[0];					      \
	  dst_ep -= 1;							      \
	  __nbytes -= 1;						      \
	  ((byte *) dst_ep)[0] = __x;					      \
	}								      \
    } while (0)

/* Copy *up to* NBYTES bytes from SRC_BP to DST_BP, with
   the assumption that DST_BP is aligned on an OPSIZ multiple.  If
   not all bytes could be easily copied, store remaining number of bytes
   in NBYTES_LEFT, otherwise store 0.  */
extern void _wordcopy_fwd_aligned (long int, long int, size_t) __THROW;
extern void _wordcopy_fwd_dest_aligned (long int, long int, size_t) __THROW;
/// 字拷贝, 注意会设置nbytes_left
/// 宏定义就是替换, 没有调用函数的过程
#define WORD_COPY_FWD(dst_bp, src_bp, nbytes_left, nbytes)		      \
  do									      \
    {									      \
      if (src_bp % OPSIZ == 0)						      \
	_wordcopy_fwd_aligned (dst_bp, src_bp, (nbytes) / OPSIZ);	      \
      else								      \
	_wordcopy_fwd_dest_aligned (dst_bp, src_bp, (nbytes) / OPSIZ);	      \
      src_bp += (nbytes) & -OPSIZ;					      \
      dst_bp += (nbytes) & -OPSIZ;					      \
      (nbytes_left) = (nbytes) % OPSIZ;					      \
    } while (0)
```

#### memcpy 内存字节拷贝

memcpy的实现比较简单, 基本思路就是从src开始按字节拷贝到destination
```cpp
void  *memcpy(void *dest, const void *src, size_t count)
{

char *p = (char *)src;

while (count--)

}

```

为了加速拷贝, 这里有个块的概念先拷贝大块, 剩余字节用小块拷贝

memcpy可能导致内存覆盖的问题, 返回目的地起始地址

```cpp
void *
memcpy (dstpp, srcpp, len)
     void *dstpp;
     const void *srcpp;
     size_t len;
{
  unsigned long int dstp = (long int) dstpp;
  unsigned long int srcp = (long int) srcpp;

  if (len >= OP_T_THRES)  // #define	OP_T_THRES	16
    {
      /* Copy just a few bytes to make DSTP aligned.  */
      /*
      #define	op_t	unsigned long int 8字节
      #define OPSIZ	(sizeof(op_t))
      */
      len -= (-dstp) % OPSIZ; // len满足word的倍数
      BYTE_COPY_FWD (dstp, srcp, (-dstp) % OPSIZ);  /// 不满足字长的先拷贝

      /* Copy whole pages from SRCP to DSTP by virtual address manipulation,
	 as much as possible.  */

      PAGE_COPY_FWD_MAYBE (dstp, srcp, len, len); /// 拷贝大块的时候会用

      /* Copy from SRCP to DSTP taking advantage of the known alignment of
	 DSTP.  Number of bytes remaining is put in the third argument,
	 i.e. in LEN.  This number may vary from machine to machine.  */

      WORD_COPY_FWD (dstp, srcp, len, len); /// 拷贝word, len变为nbytes_left

    }

  /* There are just a few bytes to copy.  Use byte memory operations.  */
  BYTE_COPY_FWD (dstp, srcp, len);  // 按字节拷贝

  return dstpp;
}


#define BYTE_COPY_FWD(dst_bp, src_bp, nbytes)				      \
  do									      \
    {									      \
      size_t __nbytes = (nbytes);					      \
      if (__nbytes & 1)							      \
        {								      \
	  ((byte *) dst_bp)[0] =  ((byte *) src_bp)[0];			      \
	  src_bp += 1;							      \
	  dst_bp += 1;							      \
	  __nbytes -= 1;						      \
        }								      \
      while (__nbytes > 0)						      \
	{								      \
	  byte __x = ((byte *) src_bp)[0];				      \
	  byte __y = ((byte *) src_bp)[1];				      \
	  src_bp += 2;							      \
	  __nbytes -= 2;						      \
	  ((byte *) dst_bp)[0] = __x;					      \
	  ((byte *) dst_bp)[1] = __y;					      \
	  dst_bp += 2;							      \
	}								      \
    } while (0)
```

#### memmove 无覆盖的内存字节拷贝

memmove不是移动, 只是在memcpy基础上, 可以防止地址段重叠前向拷贝造成的覆盖问题。memove可以防止memcpy的拷贝, 一般如果将数据拷贝到开辟的新内存, 用memcpy; 如果还是这个内存, 例如array的插入所有字符向后移动一位, 用memove。


关键区别是, memcpy用前向拷贝BYTE_COPY_FWD; 而memmove用的是后向拷贝BYTE_COPY_BWD。当内存发生局部重叠的时候，memmove保证拷贝的结果是正确的，memcpy不保证拷贝的结果的正确

```cpp
#ifndef MEMMOVE
#define MEMMOVE memmove
#endif

rettype
inhibit_loop_to_libcall
MEMMOVE (a1, a2, len)
     a1const void *a1;
     a2const void *a2;
     size_t len;
{
  unsigned long int dstp = (long int) dest;
  unsigned long int srcp = (long int) src;

  /* This test makes the forward copying code be used whenever possible.
     Reduces the working set.  */
  if (dstp - srcp >= len)	/* 地址差>len, 不会发生重叠  */
    {
      /* Copy from the beginning to the end.  */

      if (len >= OP_T_THRES)    /// 正常拷贝, 从大块到小块
	{
	  /* Copy just a few bytes to make DSTP aligned.  */
	  len -= (-dstp) % OPSIZ;
	  BYTE_COPY_FWD (dstp, srcp, (-dstp) % OPSIZ);

	  PAGE_COPY_FWD_MAYBE (dstp, srcp, len, len);

	  WORD_COPY_FWD (dstp, srcp, len, len);

	  /* Fall out and copy the tail.  */
	}

      /* There are just a few bytes to copy.  Use byte memory operations.  */
      BYTE_COPY_FWD (dstp, srcp, len);
    }
  else  // 可能发生重叠, 全部后向拷贝。(起始可以直接后向拷贝, 不用判断)
    {
      /* Copy from the end to the beginning.  */
      srcp += len;
      dstp += len;

      /* If there not too few bytes to copy, use word copy.  */
      if (len >= OP_T_THRES)
	{
	  /* Copy just a few bytes to make DSTP aligned.  */
	  len -= dstp % OPSIZ;

    /// 后向拷贝, 防止覆盖
	  BYTE_COPY_BWD (dstp, srcp, dstp % OPSIZ);

	  /* Copy from SRCP to DSTP taking advantage of the known
	     alignment of DSTP.  Number of bytes remaining is put
	     in the third argument, i.e. in LEN.  This number may
	     vary from machine to machine.  */

	  WORD_COPY_BWD (dstp, srcp, len, len);

	  /* Fall out and copy the tail.  */
	}
      /* There are just a few bytes to copy.  Use byte memory operations.  */
      BYTE_COPY_BWD (dstp, srcp, len);
    }

  RETURN (dest);
}
```

#### memset 内存字节赋值

使用方法`memset(a, 0, sizeof(a))`, 块状赋值。由于采用了内部优化, 初始化为0或者-1, 不能初始化成别的数。

```cpp
/// memset(a, 1, sizeof(a))
/// len为字节数
void *
inhibit_loop_to_libcall
memset (dstpp, c, len)
     void *dstpp;
     int c;
     size_t len;
{
  long int dstp = (long int) dstpp;

  if (len >= 8)
    {
      size_t xlen;
      op_t cccc;  /// #define	op_t	unsigned long int

        /// 赋值的块为opt, 也就是unsigned long int, 64位表示8字节
      cccc = (unsigned char) c; /// c表示一个字节， 如果c=1, cccc = 0x 00000001 四个字节
      cccc |= cccc << 8;
      cccc |= cccc << 16;
      if (OPSIZ > 4)
	/* Do the shift in two steps to avoid warning if long has 32 bits.  */
	cccc |= (cccc << 16) << 16;

      /* There are at least some bytes to set.
	 No need to test for LEN == 0 in this alignment loop.  */
      while (dstp % OPSIZ != 0)
	{
	  ((byte *) dstp)[0] = c;
	  dstp += 1;
	  len -= 1;
	}

      /* Write 8 `op_t' per iteration until less than 8 `op_t' remain.  */
      xlen = len / (OPSIZ * 8); /// unsigned long int
      while (xlen > 0)
	{
	  ((op_t *) dstp)[0] = cccc; /// 每一个代表一个unsigned long int, 都赋给一个00000c0000c0000c这种, 显然只有c=0时才能起到初始化效果
	  ((op_t *) dstp)[1] = cccc;
	  ((op_t *) dstp)[2] = cccc;
	  ((op_t *) dstp)[3] = cccc;
	  ((op_t *) dstp)[4] = cccc;
	  ((op_t *) dstp)[5] = cccc;
	  ((op_t *) dstp)[6] = cccc;
	  ((op_t *) dstp)[7] = cccc;
	  dstp += 8 * OPSIZ;
	  xlen -= 1;
	}
  /* Write the last few bytes.  */
  while (len > 0)
    {
      ((byte *) dstp)[0] = c;
      dstp += 1;
      len -= 1;
    }

  return dstpp;
}
```
